5/// This is meant to be used if there's need to infer a single expression type
6/// out of the context of a function. It will be assumed that there are no
7/// intermediate variables (`let` statements). initially, but one can still use
8/// intermediate block expression to annotate types.
9///
10/// This is useful in the context of Diesel's `Selectable` macro.
11pub fn infer_expression_type(
12    expr: &syn::Expr,
13    type_hint: Option<&syn::Type>,
14    inferrer_settings: &InferrerSettings,
15) -> (syn::Type, Vec<syn::Error>) {
16    let local_variables_map = LocalVariablesMap {
17        inferrer_settings,
18        inner: LocalVariablesMapInner {
19            map: Default::default(),
20            parent: None,
21        },
22    };
23    let inferrer = local_variables_map.inferrer();
24    let type_ = inferrer.infer_expression_type(expr, type_hint);
25
26    let errors = inferrer
27        .into_errors()
28        .into_iter()
29        .filter_map(|rc| {
30            // Extracting from the `Rc` only if it's the last reference is an elegant way to
31            // deduplicate errors For this to work it is necessary that the rest of
32            // the errors (those from the local variables map that weren't used) are
33            // dropped before, which is the case here, and that we are iterating on the
34            // errors in an owned manner.
35            Rc::try_unwrap(rc).ok()
36        })
37        .collect();
38    (type_, errors)
39}

56pub(crate) struct TypeInferrer<'s> {
57    local_variables_map: &'s LocalVariablesMap<'s, 's>,
58    errors: std::cell::RefCell<Vec<Rc<syn::Error>>>,

61impl TypeInferrer<'_> {
62    /// Calls `try_infer_expression_type` and falls back to `_` if it fails,
63    /// collecting the error for display
64    pub(crate) fn infer_expression_type(
65        &self,
66        expr: &syn::Expr,
67        type_hint: Option<&syn::Type>,
68    ) -> syn::Type {
69        let inferred = self.try_infer_expression_type(expr, type_hint);
70
71        match inferred {
72            Ok(t) => t,
73            Err(e) => self.register_error(e, expr.span()),
74        }
75    }
76
77    fn register_error(&self, error: syn::Error, infer_type_span: Span) -> syn::Type {
78        self.errors.borrow_mut().push(Rc::new(error));
79        parse_quote_spanned!(infer_type_span=> _)
80    }
81
82    fn try_infer_expression_type(
83        &self,
84        expr: &syn::Expr,
85        type_hint: Option<&syn::Type>,
86    ) -> Result<syn::Type, syn::Error> {
87        let expression_type: syn::Type = match (
88            expr,
89            type_hint.filter(|h| !matches!(h, syn::Type::Infer(_))),
90        ) {
91            (syn::Expr::Group(syn::ExprGroup { expr, .. }), type_hint) => {
92                return self.try_infer_expression_type(expr, type_hint)
93            }
94            (syn::Expr::Paren(syn::ExprParen { expr, .. }), type_hint) => {
95                return self.try_infer_expression_type(expr, type_hint)
96            }
97            (
98                syn::Expr::Tuple(syn::ExprTuple {
99                    elems: expr_elems, ..
100                }),
101                Some(syn::Type::Tuple(
102                    type_tuple @ syn::TypeTuple {
103                        elems: type_elems, ..
104                    },
105                )),
106            ) => {
107                if type_elems.len() != expr_elems.len() {
108                    return Err(syn::Error::new(
109                        type_tuple.span(),
110                        "auto_type: tuple type and its \
111                            expression have different number of elements",
112                    ));
113                }
114                syn::Type::Tuple(syn::TypeTuple {
115                    elems: type_elems
116                        .iter()
117                        .zip(expr_elems.iter())
118                        .map(|(type_, expr)| self.infer_expression_type(expr, Some(type_)))
119                        .collect(),
120                    ..type_tuple.clone()
121                })
122            }
123            (syn::Expr::Tuple(syn::ExprTuple { elems, .. }), None) => {
124                syn::Type::Tuple(syn::TypeTuple {
125                    elems: elems
126                        .iter()
127                        .map(|e| self.infer_expression_type(e, None))
128                        .collect(),
129                    paren_token: Default::default(),
130                })
131            }
132            (syn::Expr::Path(syn::ExprPath { path, .. }), None) => {
133                // This is either a local variable or we should assume that the type exists at
134                // the same path as the function (with applied casing for last segment)
135                let path_is_ident = path.get_ident();
136                if path_is_ident.is_some_and(|ident| ident == "self") {
137                    parse_quote!(Self)
138                } else if let Some(LetStatementInferredType { type_, errors }) = path_is_ident
139                    .and_then(|path_single_ident| {
140                        self.local_variables_map.inner.get(path_single_ident)
141                    })
142                {
143                    // Since we are using this type for the computation of the current type, any
144                    // errors encountered there are relevant here
145                    self.errors.borrow_mut().extend(errors.iter().cloned());
146                    type_.clone()
147                } else {
148                    syn::Type::Path(syn::TypePath {
149                        path: path.clone(),
150                        qself: None,
151                    })
152                }
153            }
154            (syn::Expr::Call(syn::ExprCall { func, args, .. }), None) => {
155                let unsupported_function_type = || {
156                    syn::Error::new_spanned(
157                        &**func,
158                        "unsupported function type for auto_type, please provide a type hint",
159                    )
160                };
161                let func_type = self.try_infer_expression_type(func, None)?;
162                // First we extract the type of the function
163                let mut type_path = match func_type {
164                    syn::Type::Path(syn::TypePath { path, .. }) => path,
165                    _ => return Err(unsupported_function_type()),
166                };
167                // Then we update the case if specified
168                if self
169                    .local_variables_map
170                    .inferrer_settings
171                    .function_types_case
172                    != Case::DoNotChange
173                {
174                    if let Some(last) = type_path.segments.last_mut() {
175                        last.ident = self
176                            .local_variables_map
177                            .inferrer_settings
178                            .function_types_case
179                            .ident_with_case(&last.ident);
180                    }
181                }
182                // Then we will add the generic arguments
183                let last_segment = type_path
184                    .segments
185                    .last_mut()
186                    .ok_or_else(unsupported_function_type)?;
187                last_segment.arguments = self.infer_generics_or_use_hints(
188                    None,
189                    args,
190                    match &last_segment.arguments {
191                        syn::PathArguments::None => None,
192                        syn::PathArguments::AngleBracketed(ab) => Some(ab),
193                        syn::PathArguments::Parenthesized(_) => {
194                            return Err(unsupported_function_type())
195                        }
196                    },
197                )?;
198                syn::Type::Path(syn::TypePath {
199                    path: type_path,
200                    qself: None,
201                })
202            }
203            (
204                syn::Expr::MethodCall(syn::ExprMethodCall {
205                    receiver,
206                    method,
207                    turbofish,
208                    args,
209                    ..
210                }),
211                None,
212            ) => syn::Type::Path(syn::TypePath {
213                path: syn::Path {
214                    segments: self
215                        .local_variables_map
216                        .inferrer_settings
217                        .dsl_path
218                        .segments
219                        .iter()
220                        .cloned()
221                        .chain([syn::PathSegment {
222                            ident: self
223                                .local_variables_map
224                                .inferrer_settings
225                                .method_types_case
226                                .ident_with_case(method),
227                            arguments: self.infer_generics_or_use_hints(
228                                Some(syn::GenericArgument::Type(
229                                    self.infer_expression_type(receiver, None),
230                                )),
231                                args,
232                                turbofish.as_ref(),
233                            )?,
234                        }])
235                        .collect(),
236                    leading_colon: None,
237                },
238                qself: None,
239            }),
240            (syn::Expr::Lit(syn::ExprLit { lit, .. }), None) => match lit {
241                syn::Lit::Str(_) => parse_quote_spanned!(lit.span()=> &'static str),
242                syn::Lit::ByteStr(_) => parse_quote_spanned!(lit.span()=> &'static [u8]),
243                syn::Lit::Byte(_) => parse_quote_spanned!(lit.span()=> u8),
244                syn::Lit::Char(_) => parse_quote_spanned!(lit.span()=> char),
245                syn::Lit::Int(lit_int) => literal_type(&lit_int.token())?,
246                syn::Lit::Float(lit_float) => literal_type(&lit_float.token())?,
247                syn::Lit::Bool(_) => parse_quote_spanned!(lit.span()=> bool),
248                _ => {
249                    return Err(syn::Error::new(
250                        lit.span(),
251                        "unsupported literal for auto_type, please provide a type hint",
252                    ))
253                }
254            },
255            (syn::Expr::Block(syn::ExprBlock { block, .. }), type_hint) => {
256                match block.stmts.last() {
257                    None
258                    | Some(
259                        syn::Stmt::Local(_) | syn::Stmt::Item(_) | syn::Stmt::Expr(_, Some(_)),
260                    ) => {
261                        // Empty blocks, local variables (`let`) and other item definition as last
262                        // statement, as well as expressions BUT with a semicolon, lead to the
263                        // block having unit type.
264                        match type_hint {
265                            Some(type_hint) => {
266                                // Prefer user-specified type hint to our own inference
267                                type_hint.clone()
268                            }
269                            None => parse_quote_spanned!(block.span()=> ()),
270                        }
271                    }
272                    Some(syn::Stmt::Expr(expr, None)) => {
273                        let local_variables_map = LocalVariablesMap {
274                            inferrer_settings: self.local_variables_map.inferrer_settings,
275                            inner: LocalVariablesMapInner {
276                                map: Default::default(),
277                                parent: Some(&self.local_variables_map.inner),
278                            },
279                        };
280                        local_variables_map.infer_block_expression_type(
281                            expr,
282                            type_hint,
283                            block,
284                            &mut self.errors.borrow_mut(),
285                        )
286                    }
287                    Some(syn::Stmt::Macro(syn::StmtMacro { mac, .. })) => {
288                        match type_hint {
289                            Some(type_hint) => {
290                                // User provided a type hint to the macro expression, we won't be
291                                // able to do any better than this
292                                type_hint.clone()
293                            }
294                            None => {
295                                return Err(syn::Error::new_spanned(
296                                    mac,
297                                    "auto_type: unsupported macro call as last statement in block, \
298                                        please provide a type hint",
299                                ));
300                            }
301                        }
302                    }
303                }
304            }
305            (syn::Expr::Binary(binary_expression), None) => {
306                let op_span = Span::mixed_site().located_at(binary_expression.op.span());
307                let trait_name = match binary_expression.op {
308                    syn::BinOp::Add(_) => "Add",
309                    syn::BinOp::Sub(_) => "Sub",
310                    syn::BinOp::Mul(_) => "Mul",
311                    syn::BinOp::Div(_) => "Div",
312                    syn::BinOp::Rem(_) => "Rem",
313                    syn::BinOp::BitXor(_) => "BitXor",
314                    syn::BinOp::BitAnd(_) => "BitAnd",
315                    syn::BinOp::BitOr(_) => "BitOr",
316                    syn::BinOp::Shl(_) => "Shl",
317                    syn::BinOp::Shr(_) => "Shr",
318                    syn::BinOp::And(_)
319                    | syn::BinOp::Or(_)
320                    | syn::BinOp::Eq(_)
321                    | syn::BinOp::Lt(_)
322                    | syn::BinOp::Le(_)
323                    | syn::BinOp::Ne(_)
324                    | syn::BinOp::Ge(_)
325                    | syn::BinOp::Gt(_) => return Ok(parse_quote!(bool)),
326                    syn::BinOp::AddAssign(_)
327                    | syn::BinOp::SubAssign(_)
328                    | syn::BinOp::MulAssign(_)
329                    | syn::BinOp::DivAssign(_)
330                    | syn::BinOp::RemAssign(_)
331                    | syn::BinOp::BitXorAssign(_)
332                    | syn::BinOp::BitAndAssign(_)
333                    | syn::BinOp::BitOrAssign(_)
334                    | syn::BinOp::ShlAssign(_)
335                    | syn::BinOp::ShrAssign(_) => return Ok(parse_quote!(())),
336                    _ => {
337                        // This is here because the `BinOp` enum is marked as #[non_exhaustive],
338                        // but in effect we really support all the variants
339                        return Err(syn::Error::new(
340                            op_span,
341                            format_args!(
342                                "unsupported binary operator for auto_type: {:?}",
343                                binary_expression.op
344                            ),
345                        ));
346                    }
347                };
348                let trait_name_ident = syn::Ident::new(trait_name, op_span);
349                let left_type = self.infer_expression_type(&binary_expression.left, None);
350                let right_type = self.infer_expression_type(&binary_expression.right, None);
351                parse_quote!(<#left_type as ::core::ops::#trait_name_ident<#right_type>>::Output)
352            }
353            (_, None) => {
354                return Err(syn::Error::new(
355                    expr.span(),
356                    "unsupported expression for auto_type, please provide a type hint",
357                ))
358            }
359            (_, Some(type_hint)) => type_hint.clone(),
360        };
361        Ok(expression_type)
362    }

364    /// `infer` is always supposed to be a syn::Type::Infer
365    fn infer_generics_or_use_hints(
366        &self,
367        add_first: Option<syn::GenericArgument>,
368        args: &syn::punctuated::Punctuated<syn::Expr, Token![,]>,
369        hint: Option<&syn::AngleBracketedGenericArguments>,
370    ) -> Result<syn::PathArguments, syn::Error> {
371        let arguments = syn::AngleBracketedGenericArguments {
372            args: add_first
373                .into_iter()
374                .chain(match hint {
375                    None => {
376                        // We should infer
377                        Either::Left(args.iter().map(|e| {
378                            syn::GenericArgument::Type(self.infer_expression_type(e, None))
379                        }))
380                    }
381                    Some(hint_or_override) => Either::Right(
382                        hint_or_override
383                            .args
384                            .iter()
385                            .zip(args.iter().map(Some).chain((0..).map(|_| None)))
386                            .map(|(hint, expr)| match (hint, expr) {
387                                (syn::GenericArgument::Type(type_hint), Some(expr)) => {
388                                    syn::GenericArgument::Type(
389                                        self.infer_expression_type(expr, Some(type_hint)),
390                                    )
391                                }
392                                (
393                                    generic_argument @ syn::GenericArgument::Type(syn::Type::Infer(
394                                        _,
395                                    )),
396                                    None,
397                                ) => syn::GenericArgument::Type(self.register_error(
398                                    syn::Error::new_spanned(
399                                        generic_argument,
400                                        "can't infer generic argument because \
401                                            there is no function argument to infer from \
402                                            (less function arguments than generic arguments)",
403                                    ),
404                                    generic_argument.span(),
405                                )),
406                                (generic_argument, _) => generic_argument.clone(),
407                            }),
408                    ),
409                })
410                .collect(),
411            colon2_token: None, // there is no colon2 in types, only in function calls
412            lt_token: Default::default(),
413            gt_token: Default::default(),
414        };
415        Ok(if arguments.args.is_empty() {
416            syn::PathArguments::None
417        } else {
418            syn::PathArguments::AngleBracketed(arguments)
419        })
420    }

427fn literal_type(t: &proc_macro2::Literal) -> Result<syn::Type, syn::Error> {
428    let val = t.to_string();
429    let type_suffix = &val[val
430        .find(|c: char| !c.is_ascii_digit() && c != '_')
431        .ok_or_else(|| {
432            syn::Error::new_spanned(
433                t,
434                format_args!("literals must have type suffix for auto_type, e.g. `{val}_i64`"),
435            )
436        })?..];
437    syn::parse_str(type_suffix)
438        .map_err(|_| syn::Error::new_spanned(t, "invalid type suffix for literal"))
439}
