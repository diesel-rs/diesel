12/* j0(x), y0(x)
13 * Bessel function of the first and second kinds of order zero.
14 * Method -- j0(x):
15 *      1. For tiny x, we use j0(x) = 1 - x^2/4 + x^4/64 - ...
16 *      2. Reduce x to |x| since j0(x)=j0(-x),  and
17 *         for x in (0,2)
18 *              j0(x) = 1-z/4+ z^2*R0/S0,  where z = x*x;
19 *         (precision:  |j0-1+z/4-z^2R0/S0 |<2**-63.67 )
20 *         for x in (2,inf)
21 *              j0(x) = sqrt(2/(pi*x))*(p0(x)*cos(x0)-q0(x)*sin(x0))
22 *         where x0 = x-pi/4. It is better to compute sin(x0),cos(x0)
23 *         as follow:
24 *              cos(x0) = cos(x)cos(pi/4)+sin(x)sin(pi/4)
25 *                      = 1/sqrt(2) * (cos(x) + sin(x))
26 *              sin(x0) = sin(x)cos(pi/4)-cos(x)sin(pi/4)
27 *                      = 1/sqrt(2) * (sin(x) - cos(x))
28 *         (To avoid cancellation, use
29 *              sin(x) +- cos(x) = -cos(2x)/(sin(x) -+ cos(x))
30 *          to compute the worse one.)
31 *
32 *      3 Special cases
33 *              j0(nan)= nan
34 *              j0(0) = 1
35 *              j0(inf) = 0
36 *
37 * Method -- y0(x):
38 *      1. For x<2.
39 *         Since
40 *              y0(x) = 2/pi*(j0(x)*(ln(x/2)+Euler) + x^2/4 - ...)
41 *         therefore y0(x)-2/pi*j0(x)*ln(x) is an even function.
42 *         We use the following function to approximate y0,
43 *              y0(x) = U(z)/V(z) + (2/pi)*(j0(x)*ln(x)), z= x^2
44 *         where
45 *              U(z) = u00 + u01*z + ... + u06*z^6
46 *              V(z) = 1  + v01*z + ... + v04*z^4
47 *         with absolute approximation error bounded by 2**-72.
48 *         Note: For tiny x, U/V = u0 and j0(x)~1, hence
49 *              y0(tiny) = u0 + (2/pi)*ln(tiny), (choose tiny<2**-27)
50 *      2. For x>=2.
51 *              y0(x) = sqrt(2/(pi*x))*(p0(x)*cos(x0)+q0(x)*sin(x0))
52 *         where x0 = x-pi/4. It is better to compute sin(x0),cos(x0)
53 *         by the method mentioned above.
54 *      3. Special cases: y0(0)=-inf, y0(x<0)=NaN, y0(inf)=0.
55 */

61/* common method when |x|>=2 */
62fn common(ix: u32, x: f64, y0: bool) -> f64 {
63    let s: f64;
64    let mut c: f64;
65    let mut ss: f64;
66    let mut cc: f64;
67    let z: f64;
68
69    /*
70     * j0(x) = sqrt(2/(pi*x))*(p0(x)*cos(x-pi/4)-q0(x)*sin(x-pi/4))
71     * y0(x) = sqrt(2/(pi*x))*(p0(x)*sin(x-pi/4)+q0(x)*cos(x-pi/4))
72     *
73     * sin(x-pi/4) = (sin(x) - cos(x))/sqrt(2)
74     * cos(x-pi/4) = (sin(x) + cos(x))/sqrt(2)
75     * sin(x) +- cos(x) = -cos(2x)/(sin(x) -+ cos(x))
76     */
77    s = sin(x);
78    c = cos(x);
79    if y0 {
80        c = -c;
81    }
82    cc = s + c;
83    /* avoid overflow in 2*x, big ulp error when x>=0x1p1023 */
84    if ix < 0x7fe00000 {
85        ss = s - c;
86        z = -cos(2.0 * x);
87        if s * c < 0.0 {
88            cc = z / ss;
89        } else {
90            ss = z / cc;
91        }
92        if ix < 0x48000000 {
93            if y0 {
94                ss = -ss;
95            }
96            cc = pzero(x) * cc - qzero(x) * ss;
97        }
98    }
99    return INVSQRTPI * cc / sqrt(x);
100}

112/// Zeroth order of the [Bessel function](https://en.wikipedia.org/wiki/Bessel_function) of the first kind (f64).
113#[cfg_attr(all(test, assert_no_panic), no_panic::no_panic)]
114pub fn j0(mut x: f64) -> f64 {
115    let z: f64;
116    let r: f64;
117    let s: f64;
118    let mut ix: u32;
119
120    ix = get_high_word(x);
121    ix &= 0x7fffffff;
122
123    /* j0(+-inf)=0, j0(nan)=nan */
124    if ix >= 0x7ff00000 {
125        return 1.0 / (x * x);
126    }
127    x = fabs(x);
128
129    if ix >= 0x40000000 {
130        /* |x| >= 2 */
131        /* large ulp error near zeros: 2.4, 5.52, 8.6537,.. */
132        return common(ix, x, false);
133    }
134
135    /* 1 - x*x/4 + x*x*R(x^2)/S(x^2) */
136    if ix >= 0x3f200000 {
137        /* |x| >= 2**-13 */
138        /* up to 4ulp error close to 2 */
139        z = x * x;
140        r = z * (R02 + z * (R03 + z * (R04 + z * R05)));
141        s = 1.0 + z * (S01 + z * (S02 + z * (S03 + z * S04)));
142        return (1.0 + x / 2.0) * (1.0 - x / 2.0) + z * (r / s);
143    }
144
145    /* 1 - x*x/4 */
146    /* prevent underflow */
147    /* inexact should be raised when x!=0, this is not done correctly */
148    if ix >= 0x38000000 {
149        /* |x| >= 2**-127 */
150        x = 0.25 * x * x;
151    }
152    return 1.0 - x;
153}

167/// Zeroth order of the [Bessel function](https://en.wikipedia.org/wiki/Bessel_function) of the second kind (f64).
168#[cfg_attr(all(test, assert_no_panic), no_panic::no_panic)]
169pub fn y0(x: f64) -> f64 {
170    let z: f64;
171    let u: f64;
172    let v: f64;
173    let ix: u32;
174    let lx: u32;
175
176    ix = get_high_word(x);
177    lx = get_low_word(x);
178
179    /* y0(nan)=nan, y0(<0)=nan, y0(0)=-inf, y0(inf)=0 */
180    if ((ix << 1) | lx) == 0 {
181        return -1.0 / 0.0;
182    }
183    if (ix >> 31) != 0 {
184        return 0.0 / 0.0;
185    }
186    if ix >= 0x7ff00000 {
187        return 1.0 / x;
188    }
189
190    if ix >= 0x40000000 {
191        /* x >= 2 */
192        /* large ulp errors near zeros: 3.958, 7.086,.. */
193        return common(ix, x, true);
194    }
195
196    /* U(x^2)/V(x^2) + (2/pi)*j0(x)*log(x) */
197    if ix >= 0x3e400000 {
198        /* x >= 2**-27 */
199        /* large ulp error near the first zero, x ~= 0.89 */
200        z = x * x;
201        u = U00 + z * (U01 + z * (U02 + z * (U03 + z * (U04 + z * (U05 + z * U06)))));
202        v = 1.0 + z * (V01 + z * (V02 + z * (V03 + z * V04)));
203        return u / v + TPI * (j0(x) * log(x));
204    }
205    return U00 + TPI * log(x);
206}

285fn pzero(x: f64) -> f64 {
286    let p: &[f64; 6];
287    let q: &[f64; 5];
288    let z: f64;
289    let r: f64;
290    let s: f64;
291    let mut ix: u32;
292
293    ix = get_high_word(x);
294    ix &= 0x7fffffff;
295    if ix >= 0x40200000 {
296        p = &PR8;
297        q = &PS8;
298    } else if ix >= 0x40122E8B {
299        p = &PR5;
300        q = &PS5;
301    } else if ix >= 0x4006DB6D {
302        p = &PR3;
303        q = &PS3;
304    } else
305    /*ix >= 0x40000000*/
306    {
307        p = &PR2;
308        q = &PS2;
309    }
310    z = 1.0 / (x * x);
311    r = p[0] + z * (p[1] + z * (p[2] + z * (p[3] + z * (p[4] + z * p[5]))));
312    s = 1.0 + z * (q[0] + z * (q[1] + z * (q[2] + z * (q[3] + z * q[4]))));
313    return 1.0 + r / s;
314}

393    2.12666388511798828631e+02,  /* 0x406A9553, 0x0E001365 */
394    -5.31095493882666946917e+00, /* 0xC0153E6A, 0xF8B32931 */
395];
396
397fn qzero(x: f64) -> f64 {
398    let p: &[f64; 6];
399    let q: &[f64; 6];
400    let s: f64;
401    let r: f64;
402    let z: f64;
403    let mut ix: u32;
404
405    ix = get_high_word(x);
406    ix &= 0x7fffffff;
407    if ix >= 0x40200000 {
408        p = &QR8;
409        q = &QS8;
410    } else if ix >= 0x40122E8B {
411        p = &QR5;
412        q = &QS5;
413    } else if ix >= 0x4006DB6D {
414        p = &QR3;
415        q = &QS3;
416    } else
417    /*ix >= 0x40000000*/
418    {
419        p = &QR2;
420        q = &QS2;
421    }
422    z = 1.0 / (x * x);
423    r = p[0] + z * (p[1] + z * (p[2] + z * (p[3] + z * (p[4] + z * p[5]))));
424    s = 1.0 + z * (q[0] + z * (q[1] + z * (q[2] + z * (q[3] + z * (q[4] + z * q[5])))));
425    return (-0.125 + r / s) / x;
426}
