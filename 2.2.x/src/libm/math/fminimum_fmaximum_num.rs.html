1/// Return the lesser of two arguments or, if either argument is NaN, NaN.
2///
3/// This coincides with IEEE 754-2019 `minimumNumber`. The result orders -0.0 < 0.0.
4#[cfg(f16_enabled)]
5#[cfg_attr(all(test, assert_no_panic), no_panic::no_panic)]
6pub fn fminimum_numf16(x: f16, y: f16) -> f16 {
7    super::generic::fminimum_num(x, y)
8}

74    fn fminimum_num_spec_test<F: Float>(f: impl Fn(F, F) -> F) {
75        let cases = [
76            (F::ZERO, F::ZERO, F::ZERO),
77            (F::ONE, F::ONE, F::ONE),
78            (F::ZERO, F::ONE, F::ZERO),
79            (F::ONE, F::ZERO, F::ZERO),
80            (F::ZERO, F::NEG_ONE, F::NEG_ONE),
81            (F::NEG_ONE, F::ZERO, F::NEG_ONE),
82            (F::INFINITY, F::ZERO, F::ZERO),
83            (F::NEG_INFINITY, F::ZERO, F::NEG_INFINITY),
84            (F::NAN, F::ZERO, F::ZERO),
85            (F::ZERO, F::NAN, F::ZERO),
86            (F::NAN, F::NAN, F::NAN),
87            (F::ZERO, F::NEG_ZERO, F::NEG_ZERO),
88            (F::NEG_ZERO, F::ZERO, F::NEG_ZERO),
89        ];
90
91        for (x, y, res) in cases {
92            let val = f(x, y);
93            assert_biteq!(val, res, "fminimum_num({}, {})", Hexf(x), Hexf(y));
94        }
95    }
