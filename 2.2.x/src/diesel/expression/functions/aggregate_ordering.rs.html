4define_sql_function! {

42mod private {
43    use crate::sql_types::{IntoNullable, SingleValue, SqlOrd, SqlType};
44    pub trait SqlOrdAggregate: SingleValue {
45        type Ret: SqlType + SingleValue;
46    }

48    impl<T> SqlOrdAggregate for T
