9macro_rules! impl_from_int_primitive {
10    ($t:ty) => {
11        impl From<$t> for BigDecimal {
12            fn from(n: $t) -> Self {
13                BigDecimal {
14                    int_val: n.into(),
15                    scale: 0,
16                }
17            }
18        }
19
20        impl From<&$t> for BigDecimal {
21            fn from(n: &$t) -> Self {
22                BigDecimal {
23                    int_val: (*n).into(),
24                    scale: 0,
25                }
26            }
27        }
28    };
29}
