Function dotenvy::from_read_iter

pub fn from_read_iter<R: Read>(reader: R) -> Iter<R> ⓘ
Expand description

Returns an iterator over environment variables from io::Read.

§Examples

use std::io::Read;
use std::os::unix::net::UnixStream;

let mut stream = UnixStream::connect("/some/socket")?;

for item in dotenvy::from_read_iter(stream) {
    let (key, val) = item?;
    println!("{}={}", key, val);
}
