20macro_rules! with_original {
21    ($trayt:ident, $func:ident, $syn:path) => {
22        impl<T: $trayt> $trayt for WithOriginal<T, $syn> {
23            fn $func(value: &$syn) -> Result<Self> {
24                Ok(WithOriginal::new($trayt::$func(value)?, value.clone()))
25            }
