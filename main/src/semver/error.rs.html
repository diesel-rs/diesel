32impl Display for Error {
33    fn fmt(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
34        match &self.kind {
35            ErrorKind::Empty => formatter.write_str("empty string, expected a semver version"),
36            ErrorKind::UnexpectedEnd(pos) => {
37                formatter.write_fmt(format_args!("unexpected end of input while parsing {0}",
        pos))write!(formatter, "unexpected end of input while parsing {}", pos)
38            }
39            ErrorKind::UnexpectedChar(pos, ch) => {
40                formatter.write_fmt(format_args!("unexpected character {0} while parsing {1}",
        QuotedChar(*ch), pos))write!(
41                    formatter,

44                    pos,
45                )
46            }
47            ErrorKind::UnexpectedCharAfter(pos, ch) => {
48                formatter.write_fmt(format_args!("unexpected character {0} after {1}",
        QuotedChar(*ch), pos))write!(
49                    formatter,

51                    QuotedChar(*ch),
52                    pos,
53                )
54            }

        QuotedChar(*ch)))write!(
57                    formatter,

62            }

64                formatter.write_fmt(format_args!("invalid leading zero in {0}", pos))write!(formatter, "invalid leading zero in {}", pos)
65            }
66            ErrorKind::Overflow(pos) => {

113struct QuotedChar(char);

115impl Display for QuotedChar {
116    fn fmt(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
117        // Standard library versions prior to https://github.com/rust-lang/rust/pull/95345
118        // print character 0 as '\u{0}'. We prefer '\0' to keep error messages
119        // the same across all supported Rust versions.
120        if self.0 == '\0' {
121            formatter.write_str("'\\0'")
122        } else {
123            formatter.write_fmt(format_args!("{0:?}", self.0))write!(formatter, "{:?}", self.0)
124        }
125    }
