51    /// Gets the name of the table, as specified on creation.
52    ///
53    /// # Example
54    ///
55    /// ```rust
56    /// use diesel_dynamic_schema::{Table, Schema};
57    /// let table: Table<&str> = "users".into();
58    /// assert_eq!(table.name(), &"users");
59    /// let schema: Schema<&str> = "public".into();
60    /// let table_with_schema: Table<&str> = schema.table("posts");
61    /// assert_eq!(table_with_schema.name(), &"posts");
62    /// ```
