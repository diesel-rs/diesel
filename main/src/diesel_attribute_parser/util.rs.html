5pub fn unknown_attribute(name: &Ident, valid: &[&str]) -> syn::Error {
6    let prefix = if valid.len() == 1 { "" } else { " one of" };
7
8    syn::Error::new(
9        name.span(),
10        ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("unknown attribute, expected{1} `{0}`",
                valid.join("`, `"), prefix))
    })format!(
11            "unknown attribute, expected{prefix} `{}`",
12            valid.join("`, `")
13        ),
14    )
15}

17pub fn parse_eq<T: Parse>(input: ParseStream, help: &str) -> Result<T> {
18    if input.is_empty() {
19        return Err(syn::Error::new(
20            input.span(),
21            ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("unexpected end of input, expected `=`\nhelp: the correct format looks like `#[diesel({0})]`",
                help))
    })format!(
22                "unexpected end of input, expected `=`\n\
23                 help: the correct format looks like `#[diesel({help})]`",
24            ),
25        ));
26    }
27
28    input.parse::<Eq>()?;
29    input.parse()
30}

51pub fn parse_paren<T: Parse>(input: ParseStream, help: &str) -> Result<T> {
52    if input.is_empty() {
53        return Err(syn::Error::new(
54            input.span(),
55            ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("unexpected end of input, expected parentheses\nhelp: the correct format looks like `#[diesel({0})]`",
                help))
    })format!(
56                "unexpected end of input, expected parentheses\n\
57                 help: the correct format looks like `#[diesel({help})]`",
58            ),
59        ));
60    }
61
62    let content;
63    match ::syn::__private::parse_parens(&input) {
    ::syn::__private::Ok(parens) => {
        content = parens.content;
        _ = content;
        parens.token
    }
    ::syn::__private::Err(error) => { return ::syn::__private::Err(error); }
};parenthesized!(content in input);
64    content.parse()
65}
