3/// The map itself, + some settings to run the inferrer with
4pub(crate) struct LocalVariablesMap<'a, 'p> {
5    pub(crate) inferrer_settings: &'a InferrerSettings,
6    /// The map, with an optional parent (to support nested blocks)
7    pub(crate) inner: LocalVariablesMapInner<'a, 'p>,
8}
9/// The map, with an optional parent (to support nested blocks)
10pub(crate) struct LocalVariablesMapInner<'a, 'p> {
11    pub(crate) map: HashMap<&'a Ident, LetStatementInferredType>,
12    pub(crate) parent: Option<&'p LocalVariablesMapInner<'a, 'p>>,
13}
14pub(crate) struct LetStatementInferredType {
15    pub(crate) type_: Type,
16    pub(crate) errors: Vec<Rc<syn::Error>>,
17}
18impl LocalVariablesMapInner<'_, '_> {
19    /// Lookup in this map, and if not found, in the parent map
20    /// This is to support nested blocks
21    pub(crate) fn get(&self, ident: &Ident) -> Option<&LetStatementInferredType> {
22        match self.map.get(ident) {
23            Some(inferred_type) => Some(inferred_type),
24            None => match self.parent {
25                Some(parent) => parent.get(ident),
26                None => None,
27            },
28        }
29    }
30}

32impl<'a> LocalVariablesMap<'a, '_> {
33    pub(crate) fn process_pat(
34        &mut self,
35        pat: &'a syn::Pat,
36        type_ascription: Option<&'a Type>,
37        local_init_expression: Option<&'a syn::Expr>,
38    ) -> Result<(), syn::Error> {
39        // Either the let statement hints the type or we have to infer it
40        // Either the let statement is a simple assignment or a destructuring assignment
41        match pat {
42            syn::Pat::Type(pat_type) => {
43                self.process_pat(
44                    &pat_type.pat,
45                    Some(match type_ascription {
46                        None => &pat_type.ty,
47                        Some(type_ascription) => {
48                            return Err(syn::Error::new(
49                                type_ascription.span(),
50                                "auto_type: unexpected double type ascription",
51                            ));
52                        }
53                    }),
54                    local_init_expression,
55                )?;
56            }
57            syn::Pat::Ident(pat_ident) => {
58                self.inner.map.insert(
59                    &pat_ident.ident,
60                    match (type_ascription, local_init_expression) {
61                        (opt_type_ascription, Some(expr)) => {
62                            let inferrer = self.inferrer();
63                            LetStatementInferredType {
64                                type_: inferrer.infer_expression_type(expr, opt_type_ascription),
65                                errors: inferrer.into_errors(),
66                            }
67                        }
68                        (Some(type_ascription), None) => LetStatementInferredType {
69                            type_: type_ascription.clone(),
70                            errors: Vec::new(),
71                        },
72                        (None, None) => LetStatementInferredType {
73                            type_: ::syn::__private::parse_quote({
        let mut _s = ::quote::__private::TokenStream::new();
        let _span: ::quote::__private::Span =
            ::quote::__private::get_span(pat_ident.span()).__into_span();
        ::quote::__private::push_underscore_spanned(&mut _s, _span);
        _s
    })parse_quote_spanned!(pat_ident.span()=> _),
74                            errors: <[_]>::into_vec(::alloc::boxed::box_new([Rc::new(syn::Error::new_spanned(pat_ident,
                        "auto_type: Let statement with no type ascription \
                                    and no initializer expression is not supported"))]))vec![Rc::new(syn::Error::new_spanned(
75                                pat_ident,
76                                "auto_type: Let statement with no type ascription \
77                                    and no initializer expression is not supported",
78                            ))],
79                        },
80                    },
81                );
82            }
83            syn::Pat::Tuple(pat_tuple) => {
84                if let Some(type_ascription) = type_ascription
85                    && let Type::Tuple(type_tuple) = type_ascription
86                    && pat_tuple.elems.len() != type_tuple.elems.len()
87                {
88                    return Err(syn::Error::new(
89                        type_ascription.span(),
90                        "auto_type: tuple let assignment and its \
91                                    type ascription have different number of elements",
92                    ));
93                }
94                for (i, pat) in pat_tuple.elems.iter().enumerate() {
95                    self.process_pat(
96                        pat,
97                        match type_ascription {
98                            Some(Type::Tuple(type_tuple)) => Some(&type_tuple.elems[i]),
99                            _ => None,
100                        },
101                        match local_init_expression {
102                            Some(syn::Expr::Tuple(expr_tuple)) => Some(&expr_tuple.elems[i]),
103                            _ => None,
104                        },
105                    )?;
106                }
107            }
108            _ => {
109                // We won't be able to infer these, at the same time we don't
110                // want to error because there may be valid user
111                // code using these, and we won't need it if these variables
112                // are not needed to infer the type of the final expression.
113            }
114        };
115        Ok(())
116    }

118    pub(crate) fn process_const_generic(&mut self, const_generic: &'a syn::ConstParam) {
119        self.inner.map.insert(
120            &const_generic.ident,
121            LetStatementInferredType {
122                type_: const_generic.ty.clone(),
123                errors: Vec::new(),
124            },
125        );
126    }
127
128    /// Finishes a block inference for this map.
129    /// It may be initialized with `pat`s before (such as function parameters),
130    /// then this function is used to infer the type of the last expression in the block.
131    ///
132    /// It takes the last expression of the block as a `block_last_expr`
133    /// parameter, because depending on where this is called,
134    /// `return`/`let`,`function_call();` will or not be tolerated, so this is
135    /// matched before calling this function.
136    ///
137    /// Expects that the block has at least one statement (it is assumed that
138    /// the last statement is provided as `block_last_expr`)
139    pub(crate) fn infer_block_expression_type(
140        mut self,
141        block_last_expr: &'a syn::Expr,
142        block_type_hint: Option<&'a syn::Type>,
143        block: &'a syn::Block,
144        errors: &mut Vec<Rc<syn::Error>>,
145    ) -> syn::Type {
146        for statement in &block.stmts[0..block
147            .stmts
148            .len()
149            .checked_sub(1)
150            .expect("Block should have at least one statement, provided as `block_last_expr`")]
151        {
152            if let syn::Stmt::Local(local) = statement {
153                match self.process_pat(
154                    &local.pat,
155                    None,
156                    local.init.as_ref().map(|local_init| &*local_init.expr),
157                ) {
158                    Ok(()) => {}
159                    Err(e) => {
160                        errors.push(Rc::new(e));
161                    }
162                }
163            };
164        }
165
166        if errors.is_empty() {
167            // We haven't encountered any error so the local variables map is
168            // properly initialized.
169            // If there are no such "syntax errors", we may attempt parsing.
170            let inferrer = self.inferrer();
171            let block_output_type =
172                inferrer.infer_expression_type(block_last_expr, block_type_hint);
173            errors.extend(inferrer.into_errors());
174            block_output_type
175        } else {
176            // We don't attempt inference if there are already errors because `process_pat`
177            // is already pretty tolerant, and we don't want to only show these errors
178            // once another error starts happening, as they may be confusing to
179            // the user.
180            let block_span = block.span();
181            ::syn::__private::parse_quote({
        let mut _s = ::quote::__private::TokenStream::new();
        let _span: ::quote::__private::Span =
            ::quote::__private::get_span(block_span).__into_span();
        ::quote::__private::push_underscore_spanned(&mut _s, _span);
        _s
    })parse_quote_spanned!(block_span=> _)
182        }
183    }
