95    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
96        match self {
97            Self::None => "".fmt(f),
98            Self::Bool(v) => v.fmt(f),
99            Self::String(v) => v.fmt(f),
100            Self::Strings(v) => v.join(", ").fmt(f),
101            Self::StyledStr(v) => v.fmt(f),
