109/// Attempts to find a tool within an MSVC installation using the Windows
110/// registry as a point to search from.
111///
112/// The `arch_or_target` argument is the architecture or the Rust target name
113/// that the tool should work for (e.g. compile or link for). The supported
114/// architecture names are:
115/// - `"x64"` or `"x86_64"`
116/// - `"arm64"` or `"aarch64"`
117/// - `"arm64ec"`
118/// - `"x86"`, `"i586"` or `"i686"`
119/// - `"arm"` or `"thumbv7a"`
120///
121/// The `tool` argument is the tool to find. Supported tools include:
122/// - MSVC tools: `cl.exe`, `link.exe`, `lib.exe`, etc.
123/// - `MSBuild`: `msbuild.exe`
124/// - Visual Studio IDE: `devenv.exe`
125/// - Clang/LLVM tools: `clang.exe`, `clang++.exe`, `clang-*.exe`, `llvm-*.exe`, `lld.exe`, etc.
126///
127/// This function will return `None` if the tool could not be found, or it will
128/// return `Some(cmd)` which represents a command that's ready to execute the
129/// tool with the appropriate environment variables set.
130///
131/// To find MSVC tools, this function will first attempt to detect if we are
132/// running in the context of a developer command prompt, and then use the tools
133/// as found in the current `PATH`. If that fails, it will attempt to locate
134/// the newest MSVC toolset in the newest installed version of Visual Studio.
135/// To limit the search to a specific version of the MSVC toolset, set the
136/// VCToolsVersion environment variable to the desired version (e.g. "14.44.35207").
137///
138/// Note that this function always returns `None` for non-MSVC targets (if a
139/// full target name was specified).
140pub fn find(arch_or_target: &str, tool: &str) -> Option<Command> {
141    find_tool(arch_or_target, tool).map(|c| c.to_command())
142}
143
144/// Similar to the `find` function above, this function will attempt the same
145/// operation (finding a MSVC tool in a local install) but instead returns a
146/// `Tool` which may be introspected.
147pub fn find_tool(arch_or_target: &str, tool: &str) -> Option<Tool> {
148    let full_arch = if let Some((full_arch, rest)) = arch_or_target.split_once("-") {
149        // The logic is all tailored for MSVC, if the target is not that then
150        // bail out early.
151        if !rest.contains("msvc") {
152            return None;
153        }
154        full_arch
155    } else {
156        arch_or_target
157    };
158    find_tool_with_env(full_arch, tool, &StdEnvGetter)
159}

161pub fn find_tool_with_env(full_arch: &str, tool: &str, env_getter: &dyn EnvGetter) -> Option<Tool> {
162    // We only need the arch.
163    let target = TargetArch::new(full_arch)?;
164
165    // Looks like msbuild isn't located in the same location as other tools like
166    // cl.exe and lib.exe.
167    if tool.contains("msbuild") {
168        return impl_::find_msbuild(target, env_getter);
169    }
170
171    // Looks like devenv isn't located in the same location as other tools like
172    // cl.exe and lib.exe.
173    if tool.contains("devenv") {
174        return impl_::find_devenv(target, env_getter);
175    }
176
177    // Clang/LLVM isn't located in the same location as other tools like
178    // cl.exe and lib.exe.
179    if ["clang", "lldb", "llvm", "ld", "lld"]
180        .iter()
181        .any(|&t| tool.contains(t))
182    {
183        return impl_::find_llvm_tool(tool, target, env_getter);
184    }
185
186    // Ok, if we're here, now comes the fun part of the probing. Default shells
187    // or shells like MSYS aren't really configured to execute `cl.exe` and the
188    // various compiler tools shipped as part of Visual Studio. Here we try to
189    // first find the relevant tool, then we also have to be sure to fill in
190    // environment variables like `LIB`, `INCLUDE`, and `PATH` to ensure that
191    // the tool is actually usable.
192
193    impl_::find_msvc_environment(tool, target, env_getter)
194        .or_else(|| impl_::find_msvc_15plus(tool, target, env_getter))
195        .or_else(|| impl_::find_msvc_14(tool, target, env_getter))
196}

217/// Find the most recent installed version of Visual Studio
218///
219/// This is used by the cmake crate to figure out the correct
220/// generator.
221#[allow(clippy::disallowed_methods)]
222pub fn find_vs_version() -> Result<VsVers, String> {
223    fn has_msbuild_version(version: &str) -> bool {
224        impl_::has_msbuild_version(version, &StdEnvGetter)
225    }
226
227    match std::env::var("VisualStudioVersion") {
228        Ok(version) => match &version[..] {
229            "17.0" => Ok(VsVers::Vs17),
230            "16.0" => Ok(VsVers::Vs16),
231            "15.0" => Ok(VsVers::Vs15),
232            "14.0" => Ok(VsVers::Vs14),
233            vers => Err(::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("\n\nunsupported or unknown VisualStudio version: {0}\nif another version is installed consider running the appropriate vcvars script before building this crate\n",
                vers))
    })format!(
234                "\n\n\
235                 unsupported or unknown VisualStudio version: {vers}\n\
236                 if another version is installed consider running \
237                 the appropriate vcvars script before building this \
238                 crate\n\
239                 "
240            )),
241        },
242        _ => {
243            // Check for the presence of a specific registry key
244            // that indicates visual studio is installed.
245            if has_msbuild_version("17.0") {
246                Ok(VsVers::Vs17)
247            } else if has_msbuild_version("16.0") {
248                Ok(VsVers::Vs16)
249            } else if has_msbuild_version("15.0") {
250                Ok(VsVers::Vs15)
251            } else if has_msbuild_version("14.0") {
252                Ok(VsVers::Vs14)
253            } else {
254                Err("\n\n\
255                     couldn't determine visual studio generator\n\
256                     if VisualStudio is installed, however, consider \
257                     running the appropriate vcvars script before building \
258                     this crate\n\
259                     "
260                .to_string())
261            }
262        }
263    }

320        fn new(name: &[u8]) -> Option<Self> {
321            let handle = unsafe { LoadLibraryA(name.as_ptr() as _) };
322            (!handle.is_null()).then_some(Self(handle))
323        }

326        /// # SAFETY
327        ///
328        /// The caller must ensure that the function signature matches the actual function.
329        /// The easiest way to do this is to add an entry to `windows_sys_no_link.list` and use the
330        /// generated function for `func_signature`.
331        ///
332        /// The function returned cannot be used after the handle is dropped.
333        unsafe fn get_proc_address<F>(&self, name: &[u8]) -> Option<F> {
334            let symbol = GetProcAddress(self.0, name.as_ptr() as _);
335            symbol.map(|symbol| mem::transmute_copy(&symbol))
336        }

348    fn is_amd64_emulation_supported_inner() -> Option<bool> {
349        // GetMachineTypeAttributes is only available on Win11 22000+, so dynamically load it.
350        let kernel32 = LibraryHandle::new(b"kernel32.dll\0")?;
351        // SAFETY: GetMachineTypeAttributesFuncType is checked to match the real function signature.
352        let get_machine_type_attributes = unsafe {
353            kernel32
354                .get_proc_address::<GetMachineTypeAttributesFuncType>(b"GetMachineTypeAttributes\0")
355        }?;
356        let mut attributes = Default::default();
357        if unsafe { get_machine_type_attributes(IMAGE_FILE_MACHINE_AMD64, &mut attributes) } == S_OK
358        {
359            Some((attributes & UserEnabled) != 0)
360        } else {
361            Some(false)
362        }
363    }

366        // TODO: Replace with a OnceLock once MSRV is 1.70.

370        // Using Relaxed ordering since the Once is providing synchronization.

441    /// Detect the target architecture of `cl.exe` in the current path, and return `None` if this
442    /// fails for any reason.
443    fn vscmd_target_cl(env_getter: &dyn EnvGetter) -> Option<&'static str> {
444        let cl_exe = env_getter.get_env("PATH").and_then(|path| {
445            env::split_paths(&path)
446                .map(|p| p.join("cl.exe"))
447                .find(|p| p.exists())
448        })?;
449        let mut cl = Command::new(cl_exe);
450        cl.stderr(std::process::Stdio::piped())
451            .stdout(std::process::Stdio::null());
452
453        let out = cl.output().ok()?;
454        let cl_arch = out
455            .stderr
456            .split(|&b| b == b'\n' || b == b'\r')
457            .next()?
458            .rsplit(|&b| b == b' ')
459            .next()?;
460
461        match cl_arch {
462            b"x64" => Some("x64"),
463            b"x86" => Some("x86"),
464            b"ARM64" => Some("arm64"),
465            b"ARM" => Some("arm"),
466            _ => None,
467        }
468    }

470    /// Attempt to find the tool using environment variables set by vcvars.
471    pub(super) fn find_msvc_environment(
472        tool: &str,
473        target: TargetArch,
474        env_getter: &dyn EnvGetter,
475    ) -> Option<Tool> {
476        // Early return if the environment isn't one that is known to have compiler toolsets in PATH
477        // `VCINSTALLDIR` is set from vcvarsall.bat (developer command prompt)
478        // `VSTEL_MSBuildProjectFullPath` is set by msbuild when invoking custom build steps
479        // NOTE: `VisualStudioDir` used to be used but this isn't set when invoking msbuild from the commandline
480        if env_getter.get_env("VCINSTALLDIR").is_none()
481            && env_getter.get_env("VSTEL_MSBuildProjectFullPath").is_none()
482        {
483            return None;
484        }
485
486        // If the vscmd target differs from the requested target then
487        // attempt to get the tool using the VS install directory.
488        if is_vscmd_target(target, env_getter) == Some(false) {
489            // We will only get here with versions 15+.
490            let vs_install_dir: PathBuf = env_getter.get_env("VSINSTALLDIR")?.into();
491            tool_from_vs15plus_instance(tool, target, &vs_install_dir, env_getter)
492        } else {
493            // Fallback to simply using the current environment.
494            env_getter
495                .get_env("PATH")
496                .and_then(|path| {
497                    env::split_paths(&path)
498                        .map(|p| p.join(tool))
499                        .find(|p| p.exists())
500                })
501                .map(|path| Tool {
502                    tool: path,
503                    is_clang_cl: false,
504                    env: Vec::new(),
505                })
506        }
507    }

536    fn find_tool_in_vs16plus_path(
537        tool: &str,
538        target: TargetArch,
539        version: &'static str,
540        env_getter: &dyn EnvGetter,
541    ) -> Option<Tool> {
542        vs16plus_instances(target, version, env_getter)
543            .filter_map(|path| {
544                let path = path.join(tool);
545                if !path.is_file() {
546                    return None;
547                }
548                let mut tool = Tool {
549                    tool: path,
550                    is_clang_cl: false,
551                    env: Vec::new(),
552                };
553                if target == TargetArch::X64 {
554                    tool.env.push(("Platform".into(), "X64".into()));
555                }
556                if matches!(target, TargetArch::Arm64 | TargetArch::Arm64ec) {
557                    tool.env.push(("Platform".into(), "ARM64".into()));
558                }
559                Some(tool)
560            })
561            .next()

687        };
688        Some([
689            get_next_number()?,
690            get_next_number()?,
691            get_next_number()?,
692            get_next_number()?,
693        ])

760        tool: &str,
761        target: TargetArch,
762        instance_path: &Path,

792    fn vs15plus_vc_paths(
793        target_arch: TargetArch,
794        instance_path: &Path,
795        env_getter: &dyn EnvGetter,
796    ) -> Option<(PathBuf, PathBuf, PathBuf, PathBuf, Option<PathBuf>, PathBuf)> {
797        let version = vs15plus_vc_read_version(instance_path, env_getter)?;
798
799        let hosts = match host_arch() {
800            X86 => &["X86"],
801            X86_64 => &["X64"],
802            // Starting with VS 17.4, there is a natively hosted compiler on ARM64:
803            // https://devblogs.microsoft.com/visualstudio/arm64-visual-studio-is-officially-here/
804            // On older versions of VS, we use x64 if running under emulation is supported,
805            // otherwise use x86.
806            AARCH64 => {
807                if is_amd64_emulation_supported() {
808                    &["ARM64", "X64", "X86"][..]
809                } else {
810                    &["ARM64", "X86"]
811                }
812            }
813            _ => return None,
814        };
815        let target_dir = target_arch.as_vs_arch();
816        // The directory layout here is MSVC/bin/Host$host/$target/
817        let path = instance_path.join(r"VC\Tools\MSVC").join(version);
818        // We use the first available host architecture that can build for the target
819        let (host_path, host) = hosts.iter().find_map(|&x| {
820            let candidate = path.join("bin").join(format!("Host{}", x));
821            if candidate.join(target_dir).exists() {
822                Some((candidate, x))
823            } else {
824                None
825            }
826        })?;
827        // This is the path to the toolchain for a particular target, running
828        // on a given host
829        let bin_path = host_path.join(target_dir);
830        // But! we also need PATH to contain the target directory for the host
831        // architecture, because it contains dlls like mspdb140.dll compiled for
832        // the host architecture.
833        let host_dylib_path = host_path.join(host.to_lowercase());
834        let lib_fragment = if use_spectre_mitigated_libs(env_getter) {
835            r"lib\spectre"
836        } else {
837            "lib"
838        };
839        let lib_path = path.join(lib_fragment).join(target_dir);
840        let alt_lib_path =
841            (target_arch == TargetArch::Arm64ec).then(|| path.join(lib_fragment).join("arm64ec"));
842        let include_path = path.join("include");
843        Some((
844            path,
845            bin_path,
846            host_dylib_path,
847            lib_path,
848            alt_lib_path,
849            include_path,
850        ))
851    }

853    fn vs15plus_vc_read_version(dir: &Path, env_getter: &dyn EnvGetter) -> Option<String> {
854        if let Some(version) = env_getter.get_env("VCToolsVersion") {
855            // Restrict the search to a specific msvc version; if it doesn't exist then
856            // our caller will fail to find the tool for this instance and move on.
857            return version.to_str().map(ToString::to_string);
858        }
859
860        // Try to open the default version file.
861        let mut version_path: PathBuf =
862            dir.join(r"VC\Auxiliary\Build\Microsoft.VCToolsVersion.default.txt");
863        let mut version_file = if let Ok(f) = File::open(&version_path) {
864            f
865        } else {
866            // If the default doesn't exist, search for other version files.
867            // These are in the form Microsoft.VCToolsVersion.v143.default.txt
868            // where `143` is any three decimal digit version number.
869            // This sorts versions by lexical order and selects the highest version.
870            let mut version_file = String::new();
871            version_path.pop();
872            for file in version_path.read_dir().ok()? {
873                let name = file.ok()?.file_name();
874                let name = name.to_str()?;
875                if name.starts_with("Microsoft.VCToolsVersion.v")
876                    && name.ends_with(".default.txt")
877                    && name > &version_file
878                {
879                    version_file.replace_range(.., name);
880                }
881            }
882            if version_file.is_empty() {
883                // If all else fails, manually search for bin directories.
884                let tools_dir: PathBuf = dir.join(r"VC\Tools\MSVC");
885                return tools_dir
886                    .read_dir()
887                    .ok()?
888                    .filter_map(|file| {
889                        let file = file.ok()?;
890                        let name = file.file_name().into_string().ok()?;
891
892                        file.path().join("bin").exists().then(|| {
893                            let version = parse_version(&name);
894                            (name, version)
895                        })
896                    })
897                    .max_by(|(_, a), (_, b)| a.cmp(b))
898                    .map(|(version, _)| version);
899            }
900            version_path.push(version_file);
901            File::open(version_path).ok()?
902        };
903
904        // Get the version string from the file we found.
905        let mut version = String::new();
906        version_file.read_to_string(&mut version).ok()?;
907        version.truncate(version.trim_end().len());
908        Some(version)
909    }

1136    // When choosing the tool to use, we have to choose the one which matches
1137    // the target architecture. Otherwise we end up in situations where someone
1138    // on 32-bit Windows is trying to cross compile to 64-bit and it tries to
1139    // invoke the native 64-bit compiler which won't work.
1140    //
1141    // For the return value of this function, the first member of the tuple is
1142    // the folder of the tool we will be invoking, while the second member is
1143    // the folder of the host toolchain for that tool which is essential when
1144    // using a cross linker. We return a Vec since on x64 there are often two
1145    // linkers that can target the architecture we desire. The 64-bit host
1146    // linker is preferred, and hence first, due to 64-bit allowing it more
1147    // address space to work with and potentially being faster.
1148    fn bin_subdir(target: TargetArch) -> Vec<(&'static str, &'static str)> {
1149        match (target, host_arch()) {
1150            (TargetArch::X86, X86) => vec![("", "")],
1151            (TargetArch::X86, X86_64) => vec![("amd64_x86", "amd64"), ("", "")],
1152            (TargetArch::X64, X86) => vec![("x86_amd64", "")],
1153            (TargetArch::X64, X86_64) => vec![("amd64", "amd64"), ("x86_amd64", "")],
1154            (TargetArch::Arm, X86) => vec![("x86_arm", "")],
1155            (TargetArch::Arm, X86_64) => vec![("amd64_arm", "amd64"), ("x86_arm", "")],
1156            _ => vec![],
1157        }
1158    }

1351    // Given a registry key, look at all the sub keys and find the one which has
1352    // the maximal numeric value.
1353    //
1354    // Returns the name of the maximal key as well as the opened maximal key.
1355    fn max_version(key: &RegistryKey) -> Option<(OsString, RegistryKey)> {
1356        let mut max_vers = 0;
1357        let mut max_key = None;
1358        for subkey in key.iter().filter_map(|k| k.ok()) {
1359            let val = subkey
1360                .to_str()
1361                .and_then(|s| s.trim_start_matches('v').replace('.', "").parse().ok());
1362            let val = match val {
1363                Some(s) => s,
1364                None => continue,
1365            };
1366            if val > max_vers {
1367                if let Ok(k) = key.open(&subkey) {
1368                    max_vers = val;
1369                    max_key = Some((subkey, k));
1370                }
1371            }
1372        }
1373        max_key
1374    }

1486    /// Attempt to find the tool using environment variables set by vcvars.
1487    pub(super) fn find_msvc_environment(
1488        tool: &str,
1489        _target: TargetArch,
1490        env_getter: &dyn EnvGetter,
1491    ) -> Option<Tool> {
1492        // Early return if the environment doesn't contain a VC install.
1493        let vc_install_dir = env_getter.get_env("VCINSTALLDIR")?;
1494        let vs_install_dir = env_getter.get_env("VSINSTALLDIR")?;
1495
1496        let get_tool = |install_dir: &OsStr| {
1497            env::split_paths(install_dir)
1498                .map(|p| p.join(tool))
1499                .find(|p| p.exists())
1500                .map(|path| Tool {
1501                    tool: path,
1502                    is_clang_cl: false,
1503                    env: Vec::new(),
1504                })
1505        };
1506
1507        // Take the path of tool for the vc install directory.
1508        get_tool(vc_install_dir.as_ref())
1509            // Take the path of tool for the vs install directory.
1510            .or_else(|| get_tool(vs_install_dir.as_ref()))
1511            // Take the path of tool for the current path environment.
1512            .or_else(|| {
1513                env_getter
1514                    .get_env("PATH")
1515                    .as_ref()
1516                    .map(|path| path.as_ref())
1517                    .and_then(get_tool)
1518            })
1519    }
