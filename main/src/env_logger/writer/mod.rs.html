156fn parse_write_style(spec: &str) -> WriteStyle {
157    match spec {
158        "auto" => WriteStyle::Auto,
159        "always" => WriteStyle::Always,
160        "never" => WriteStyle::Never,
161        _ => Default::default(),
162    }
