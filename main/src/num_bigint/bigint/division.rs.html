244            self % other
245        } else if let Some(other) = other.to_i32() {

472impl Euclid for BigInt {
473    #[inline]
474    fn div_euclid(&self, v: &BigInt) -> BigInt {
475        let (q, r) = self.div_rem(v);
476        if r.is_negative() {
477            if v.is_positive() {
478                q - 1
479            } else {
480                q + 1
481            }
482        } else {
483            q
484        }
485    }
486
487    #[inline]
488    fn rem_euclid(&self, v: &BigInt) -> BigInt {
489        let r = self % v;
490        if r.is_negative() {
491            if v.is_positive() {
492                r + v
493            } else {
494                r - v
495            }
496        } else {
497            r
498        }
499    }
500
501    fn div_rem_euclid(&self, v: &Self) -> (Self, Self) {
502        let (q, r) = self.div_rem(v);
503        if r.is_negative() {
504            if v.is_positive() {
505                (q - 1, r + v)
506            } else {
507                (q + 1, r - v)
508            }
509        } else {
510            (q, r)
511        }
512    }
