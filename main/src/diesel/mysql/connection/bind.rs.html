76    pub(super) fn populate_dynamic_buffers(&mut self, stmt: &StatementUse<'_>) -> QueryResult<()> {
77        for (i, data) in self.0.data.iter_mut().enumerate() {
78            data.did_numeric_overflow_occur()?;
79            // This is safe because we are re-binding the invalidated buffers
80            // at the end of this function
81            unsafe {
82                if let Some((mut bind, offset)) = data.bind_for_truncated_data() {
83                    stmt.fetch_column(&mut bind, i, offset)?
84                } else {
85                    data.update_buffer_length()
86                }
87            }
88        }
89
90        unsafe { self.with_mysql_binds(|bind_ptr| stmt.bind_result(bind_ptr)) }
91    }

739fn known_buffer_size_for_ffi_type(tpe: ffi::enum_field_types) -> Option<usize> {
740    use self::ffi::enum_field_types as t;
741    use std::mem::size_of;
742
743    match tpe {
744        t::MYSQL_TYPE_TINY => Some(1),
745        t::MYSQL_TYPE_YEAR | t::MYSQL_TYPE_SHORT => Some(2),
746        t::MYSQL_TYPE_INT24 | t::MYSQL_TYPE_LONG | t::MYSQL_TYPE_FLOAT => Some(4),
747        t::MYSQL_TYPE_LONGLONG | t::MYSQL_TYPE_DOUBLE => Some(8),
748        t::MYSQL_TYPE_TIME
749        | t::MYSQL_TYPE_DATE
750        | t::MYSQL_TYPE_DATETIME
751        | t::MYSQL_TYPE_TIMESTAMP => Some(size_of::<MysqlTime>()),
752        _ => None,
753    }

1248    fn query_single_table(
1249        query: &'static str,
1250        conn: &MysqlConnection,
1251        bind_tpe: impl Into<(ffi::enum_field_types, Flags)>,
1252    ) -> BindData {
1253        let stmt: Statement = conn
1254            .raw_connection
1255            .prepare(query, PrepareForCache::No, &[])
1256            .unwrap();
1257        let stmt = MaybeCached::CannotCache(stmt);
1258
1259        let bind = BindData::from_tpe_and_flags(bind_tpe.into());
1260
1261        let mut binds = OutputBinds(Binds { data: vec![bind] });
1262
1263        let stmt = stmt.execute_statement(&mut binds).unwrap();
1264        stmt.populate_row_buffers(&mut binds).unwrap();
1265
1266        binds.0.data.remove(0)
1267    }

1269    fn input_bind(
1270        query: &'static str,
1271        conn: &MysqlConnection,
1272        id: i32,
1273        (field, tpe): (Vec<u8>, impl Into<(ffi::enum_field_types, Flags)>),
1274    ) {
1275        let mut stmt = conn
1276            .raw_connection
1277            .prepare(query, PrepareForCache::No, &[])
1278            .unwrap();
1279        let length = field.len() as _;
1280        let (tpe, flags) = tpe.into();
1281        let (ptr, _length, capacity) = bind_buffer(field);
1282
1283        let field_bind = BindData {
1284            tpe,
1285            bytes: ptr,
1286            capacity,
1287            length,
1288            flags,
1289            is_null: ffi::FALSE,
1290            is_truncated: None,
1291        };
1292
1293        let (ptr, length, capacity) = bind_buffer(id.to_be_bytes().to_vec());
1294
1295        let id_bind = BindData {
1296            tpe: ffi::enum_field_types::MYSQL_TYPE_LONG,
1297            bytes: ptr,
1298            capacity,
1299            length,
1300            flags: Flags::empty(),
1301            is_null: ffi::FALSE,
1302            is_truncated: None,
1303        };
1304
1305        let binds = PreparedStatementBinds(Binds {
1306            data: vec![id_bind, field_bind],
1307        });
1308        stmt.input_bind(binds).unwrap();
1309        stmt.did_an_error_occur().unwrap();
1310        let stmt = MaybeCached::CannotCache(stmt);
1311        unsafe {
1312            stmt.execute().unwrap();
1313        }
1314    }
