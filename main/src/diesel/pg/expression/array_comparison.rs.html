8/// Creates a PostgreSQL `ANY` expression.
9///
10/// As with most bare functions, this is not exported by default. You can import
11/// it specifically from `diesel::pg::expression::dsl::any`, or `diesel::dsl::any`.
12///
13/// # Example
14///
15/// ```rust
16/// # include!("../../doctest_setup.rs");
17/// # use diesel::dsl::*;
18/// #
19/// # fn main() {
20/// #     use schema::users::dsl::*;
21/// #     let connection = &mut establish_connection();
22/// #     diesel::sql_query("INSERT INTO users (name) VALUES ('Jim')").execute(connection).unwrap();
23/// let sean = (1, "Sean".to_string());
24/// let jim = (3, "Jim".to_string());
25/// let data = users.filter(name.eq(any(vec!["Sean", "Jim"])));
26/// assert_eq!(Ok(vec![sean, jim]), data.load(connection));
27/// # }
28/// ```
29#[deprecated(since = "2.0.0", note = "Use `ExpressionMethods::eq_any` instead")]
30pub fn any<ST, T>(vals: T) -> Any<T::Expression>
31where
32    T: AsArrayExpression<ST>,
33{
34    Any::new(vals.as_expression())
35}

133/// Deprecated trait used for implementing `any` and `all` (which are themselves deprecated).
134///
135/// It has several quirks:
136/// - `All<Expr: Expression<SqlType = Array<ST>>>` pretends to be an expression of type `ST`,
137///   but it's in fact some custom that can really be only used in combination with the `=`
138///   operator in some select places.
139/// - Implementations that use Subelect below are lying: they pretend to be expressions of type
140///   `Array<ST>`, but they're actually subselects, which are processed differently by Postgres
141///   and may result in different query plans.
142///   The `IntoArrayExpression` trait represents this more accurately, actually building an
143///   actual expression of type array from a subselect (by wrapping it in `ARRAY(subselect)`).
144pub trait AsArrayExpression<ST: 'static> {
145    type Expression: Expression<SqlType = Array<ST>>;
146
147    // This method is part of the public API
148    // we won't change it to appease a clippy lint
149    #[allow(clippy::wrong_self_convention)]
150    fn as_expression(self) -> Self::Expression;
151}
152
153impl<ST, T> AsArrayExpression<ST> for T
154where
155    ST: 'static,
156    T: AsExpression<Array<ST>>,
157{
158    type Expression = <T as AsExpression<Array<ST>>>::Expression;
159
160    fn as_expression(self) -> Self::Expression {
161        <T as AsExpression<Array<ST>>>::as_expression(self)
162    }
163}
