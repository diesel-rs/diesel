8fn pg_epoch() -> SystemTime {
9    let thirty_years = Duration::from_secs(946_684_800);
10    UNIX_EPOCH + thirty_years
11}
12
13#[cfg(feature = "postgres_backend")]
14impl ToSql<sql_types::Timestamp, Pg> for SystemTime {
15    fn to_sql<'b>(&'b self, out: &mut Output<'b, '_, Pg>) -> serialize::Result {
16        let (before_epoch, duration) = match self.duration_since(pg_epoch()) {
17            Ok(duration) => (false, duration),
18            Err(time_err) => (true, time_err.duration()),
19        };
20        let time_since_epoch = if before_epoch {
21            -(i64::try_from(duration_to_usecs(duration))?)
22        } else {
23            duration_to_usecs(duration).try_into()?
24        };
25        ToSql::<sql_types::BigInt, Pg>::to_sql(&time_since_epoch, &mut out.reborrow())
26    }
27}

29#[cfg(feature = "postgres_backend")]
30impl FromSql<sql_types::Timestamp, Pg> for SystemTime {
31    fn from_sql(bytes: PgValue<'_>) -> deserialize::Result<Self> {
32        let usecs_passed = <i64 as FromSql<sql_types::BigInt, Pg>>::from_sql(bytes)?;
33        let before_epoch = usecs_passed < 0;
34        let time_passed = usecs_to_duration(usecs_passed.unsigned_abs());
35
36        if before_epoch {
37            Ok(pg_epoch() - time_passed)
38        } else {
39            Ok(pg_epoch() + time_passed)
40        }
41    }
42}
43
44const USEC_PER_SEC: u64 = 1_000_000;
45const NANO_PER_USEC: u32 = 1_000;
46
47fn usecs_to_duration(usecs_passed: u64) -> Duration {
48    let seconds = usecs_passed / USEC_PER_SEC;
49    let subsecond_usecs = usecs_passed % USEC_PER_SEC;
50    let subseconds = subsecond_usecs as u32 * NANO_PER_USEC;
51    Duration::new(seconds, subseconds)
52}
53
54fn duration_to_usecs(duration: Duration) -> u64 {
55    let seconds = duration.as_secs() * USEC_PER_SEC;
56    let subseconds = duration.subsec_micros();
57    seconds + u64::from(subseconds)
58}
