121impl<S, TSM> QueryFragment<Pg> for Tablesample<S, TSM>
122where
123    S: QueryFragment<Pg>,
124    TSM: TablesampleMethod,
125{
126    fn walk_ast<'b>(&'b self, mut out: AstPass<'_, 'b, Pg>) -> QueryResult<()> {
127        self.source.walk_ast(out.reborrow())?;
128        out.push_sql(" TABLESAMPLE ");
129        out.push_sql(TSM::method_name_sql());
130        out.push_sql("(");
131        out.push_bind_param::<SmallInt, _>(&self.portion)?;
132        out.push_sql(")");
133        if let Some(f) = &self.seed {
134            out.push_sql(" REPEATABLE(");
135            out.push_bind_param::<Double, _>(f)?;
136            out.push_sql(")");
137        }
138        Ok(())
139    }
140}

200    macro_rules! assert_sql {
201        ($query:expr, $sql:expr) => {
202            let mut query_builder = <Pg as Backend>::QueryBuilder::default();
203            $query.to_sql(&mut query_builder, &Pg).unwrap();
204            let sql = query_builder.finish();
205            assert_eq!(sql, $sql);
206        };
207    }

212            name -> VarChar,
213        }

216    #[diesel_test_helper::test]
217    fn test_generated_tablesample_sql() {
218        assert_sql!(
219            users::table.tablesample_bernoulli(10),
220            "\"users\" TABLESAMPLE BERNOULLI($1)"
221        );

223        assert_sql!(
224            users::table.tablesample_system(10),
225            "\"users\" TABLESAMPLE SYSTEM($1)"
226        );

228        assert_sql!(

230            "\"users\" TABLESAMPLE SYSTEM($1) REPEATABLE($2)"
