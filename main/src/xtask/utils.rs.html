3pub fn get_exclude_for_backend<'a>(
4    backend: &str,
5    metadata: &'a Metadata,
6    wasm: bool,
7) -> Vec<&'a str> {
8    let examples = metadata.workspace_root.join("examples");
9    let backend_examples = examples.join(backend);
10    let mut out = metadata
11        .workspace_packages()
12        .into_iter()
13        .filter_map(move |p| {
14            if p.manifest_path.starts_with(&examples)
15                && !p.manifest_path.starts_with(&backend_examples)
16            {
17                Some(["--exclude", &p.name])
18            } else {
19                None
20            }
21        })
22        .flatten()
23        .collect::<Vec<_>>();
24    if wasm {
25        let additional_excludes = [
26            // command line tool is not helpful
27            // with the wasm32-unknown-unknown target
28            "diesel_cli",
29            // these pull in libsqlite3-sys
30            "getting_started_step_1_sqlite",
31            "getting_started_step_2_sqlite",
32            "getting_started_step_3_sqlite",
33            "all_about_inserts_sqlite",
34            "relations_sqlite",
35            // needs to be tested in a separate step
36            // due to broken cargo workspace feature unification
37            "sqlite-wasm-example",
38        ];
39        out.extend(
40            additional_excludes
41                .into_iter()
42                .flat_map(|v| ["--exclude", v]),
43        );
44    }
45    out
46}
