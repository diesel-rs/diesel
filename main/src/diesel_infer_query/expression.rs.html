5use crate::error::{Error, Result};
6use crate::query_source::QuerySource;
7use crate::select::SelectFieldKind;
8use sqlparser::ast::{Expr, Value};
9use std::collections::HashMap;
10
11pub(crate) fn infer_expr(
12    expr: &sqlparser::ast::Expr,
13    query_source_lookup: &HashMap<&str, QuerySource>,
14) -> Result<SelectFieldKind> {
15    match expr {
16        Expr::Value(v) => Ok(SelectFieldKind::Literal {
17            value: v.value.clone().into_string(),
18            is_null: #[allow(non_exhaustive_omitted_patterns)] match v.value {
    Value::Null => true,
    _ => false,
}matches!(v.value, Value::Null),
19        }),
20        Expr::Identifier(id) => {
21            if query_source_lookup.len() == 1 {
22                let table = query_source_lookup
23                    .values()
24                    .next()
25                    .expect("We checked there is only one value");
26                Ok(SelectFieldKind::Field {
27                    schema: table.schema.map(|s| s.to_owned()),
28                    query_source: table.name.to_owned(),
29                    field_name: id.value.clone(),
30                    // no joins here so we should be fine
31                    via_left_join: false,
32                })
33            } else {
34                Err(Error::UnsupportedSql {
35                    msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Queries without more than one table are not supported yet: `{0}`",
                id))
    })format!(
36                        "Queries without more than one table are not supported yet: `{id}`"
37                    ),
38                })
39            }
40        }
41        s @ Expr::CompoundIdentifier(ids) => match ids.as_slice() {
42            [table, field] => {
43                let table = query_source_lookup
44                    .get(table.value.as_str())
45                    .ok_or_else(|| Error::InvalidQuerySource {
46                        query_source: table.value.clone(),
47                    })?;
48                // lookup if this field comes in via a `LEFT JOIN` somewhere in the chain
49                let via_left_join = table.contains_left_join(&query_source_lookup)?;
50                Ok(SelectFieldKind::Field {
51                    schema: table.schema.map(|s| s.to_owned()),
52                    query_source: table.name.to_owned(),
53                    field_name: field.value.clone(),
54                    via_left_join,
55                })
56            }
57            _ => Err(Error::UnsupportedSql {
58                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Queries with more than 2 identifier segments are not supported yet: `{0}`",
                s))
    })format!(
59                    "Queries with more than 2 identifier segments are not supported yet: `{s}`"
60                ),
61            }),
62        },
63        Expr::Cast {
64            expr, data_type, ..
65        } => {
66            let inner = infer_expr(expr, query_source_lookup)?;
67            Ok(SelectFieldKind::Cast {
68                inner: Box::new(inner),
69                tpe: data_type.to_string(),
70            })
71        }
72        // other kinds of expressions still need to be supported
73        _e => Ok(SelectFieldKind::Unknown),
74    }
75}
