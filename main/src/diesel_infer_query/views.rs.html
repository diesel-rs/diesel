17impl ViewData {

23    /// Infer the nullablity of all VIEW fields
24    ///
25    /// This function returns a vector of optional booleans.
26    /// The number and order of elements in this vector corresponds to
27    /// the number and order of fields returned by this VIEW.
28    ///
29    /// Each value indicates whether the field is nullable (`Some(true)`),
30    /// not nullable (`Some(false)`) or if the nullablity could
31    /// not be inferred (`None`)
32    ///
33    /// This method accepts a generic [`SchemaResolver`]
34    /// to query information about relations used in this
35    /// view definition
36    pub fn infer_nullability(
37        &self,
38        resolver: &mut dyn SchemaResolver,
39    ) -> Result<Vec<Option<bool>>> {
40        self.fields
41            .iter()
42            .map(|f| f.infer_nullability(resolver))
43            .collect()
44    }
45}

47/// Infer information about a given view definition
48///
49/// This method accepts both `CREATE VIEW xyz AS SELECT …` and
50/// plain `SELECT …` statements as view definition.
51pub fn parse_view_def(definition: &str) -> Result<ViewData> {
52    let dialect = sqlparser::dialect::SQLiteDialect {};
53    let options = ParserOptions::new();
54
55    let stmt = sqlparser::parser::Parser::new(&dialect)
56        .with_options(options)
57        .try_with_sql(definition)?
58        .parse_statement()?;
59
60    let select = match stmt {
61        sqlparser::ast::Statement::Query(query) => query,
62        sqlparser::ast::Statement::CreateView { query, .. } => query,
63        stmt => {
64            return Err(Error::UnsupportedSql {
65                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unexpected statement: `{0}`",
                stmt))
    })format!("Unexpected statement: `{stmt}`"),
66            });
67        }
68    };
69    let result = match &*select.body {
70        sqlparser::ast::SetExpr::Select(select) => infer_from_select(select),
71        // we likely want to support more complex queries here as well (UNION, CTE, etc)
72        s => {
73            return Err(Error::UnsupportedSql {
74                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported query kind: `{0}`", s))
    })format!("Unsupported query kind: `{s}`"),
75            });
76        }
77    };
78    Ok(ViewData { fields: result? })
79}
