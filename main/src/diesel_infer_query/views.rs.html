19impl ViewData {
20    /// The number of fields returned by this VIEW
21    pub fn field_count(&self) -> usize {
22        self.fields.len()
23    }

25    /// Infer the nullablity of all VIEW fields
26    ///
27    /// This function returns a vector of optional booleans.
28    /// The number and order of elements in this vector corresponds to
29    /// the number and order of fields returned by this VIEW.
30    ///
31    /// Each value indicates whether the field is nullable (`Some(true)`),
32    /// not nullable (`Some(false)`) or if the nullablity could
33    /// not be inferred (`None`)
34    ///
35    /// This method accepts a generic [`SchemaResolver`]
36    /// to query information about relations used in this
37    /// view definition
38    pub fn infer_nullability(
39        &self,
40        resolver: &mut dyn SchemaResolver,
41    ) -> Result<Vec<Option<bool>>> {
42        self.fields
43            .iter()
44            .map(|f| f.infer_nullability(resolver))
45            .collect()
46    }
47
48    /// Resolve references to wildcard expressions given the provided schema resolver
49    ///
50    /// This needs to be called before any other operation is performed with this view definition
51    pub fn resolve_references(&mut self, resolver: &mut dyn SchemaResolver) -> Result<()> {
52        let fields = std::mem::take(&mut self.fields);
53        self.fields.reserve(fields.len());
54        for f in fields {
55            if let Expression::Wildcard {
56                schema,
57                relation,
58                is_left_joined,
59            } = &f.kind
60            {
61                let resolved_fields = resolver
62                    .list_fields(schema.as_deref(), relation)
63                    .map_err(|e| Error::ResolverFailure { inner: e })?;
64                for f in resolved_fields {
65                    self.fields.push(SelectField {
66                        ident: None,
67                        kind: Expression::Field {
68                            schema: schema.clone(),
69                            query_source: relation.clone(),
70                            field_name: f.name().to_owned(),
71                            via_left_join: *is_left_joined,
72                        },
73                    });
74                }
75            } else {
76                self.fields.push(f);
77            }
78        }
79        Ok(())
80    }
81}

83/// Infer information about a given view definition
84///
85/// This method accepts both `CREATE VIEW xyz AS SELECT …` and
86/// plain `SELECT …` statements as view definition.
87pub fn parse_view_def(definition: &str) -> Result<ViewData> {
88    let dialect = sqlparser::dialect::SQLiteDialect {};
89    let options = ParserOptions::new();
90
91    let stmt = sqlparser::parser::Parser::new(&dialect)
92        .with_options(options)
93        .try_with_sql(definition)?
94        .parse_statement()?;
95
96    let select = match stmt {
97        sqlparser::ast::Statement::Query(query) => query,
98        sqlparser::ast::Statement::CreateView(CreateView { query, .. }) => query,
99        stmt => {
100            return Err(Error::UnsupportedSql {
101                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unexpected statement: `{0}`",
                stmt))
    })format!("Unexpected statement: `{stmt}`"),
102            });
103        }
104    };
105    let result = match &*select.body {
106        sqlparser::ast::SetExpr::Select(select) => infer_from_select(select),
107        // we likely want to support more complex queries here as well (UNION, CTE, etc)
108        s => {
109            return Err(Error::UnsupportedSql {
110                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported query kind: `{0}`", s))
    })format!("Unsupported query kind: `{s}`"),
111            });
112        }
113    };
114    Ok(ViewData { fields: result? })
