18impl ViewData {
19    /// The number of fields returned by this VIEW
20    pub fn field_count(&self) -> usize {
21        self.fields.len()
22    }
23
24    /// Infer the nullablity of all VIEW fields
25    ///
26    /// This function returns a vector of optional booleans.
27    /// The number and order of elements in this vector corresponds to
28    /// the number and order of fields returned by this VIEW.
29    ///
30    /// Each value indicates whether the field is nullable (`Some(true)`),
31    /// not nullable (`Some(false)`) or if the nullablity could
32    /// not be inferred (`None`)
33    ///
34    /// This method accepts a generic [`SchemaResolver`]
35    /// to query information about relations used in this
36    /// view definition
37    pub fn infer_nullability(
38        &self,
39        resolver: &mut dyn SchemaResolver,
40    ) -> Result<Vec<Option<bool>>> {
41        self.fields
42            .iter()
43            .map(|f| f.infer_nullability(resolver))
44            .collect()
45    }
46
47    /// Resolve references to wildcard expressions given the provided schema resolver
48    ///
49    /// This needs to be called before any other operation is performed with this view definition
50    pub fn resolve_references(&mut self, resolver: &mut dyn SchemaResolver) -> Result<()> {
51        let fields = std::mem::take(&mut self.fields);
52        self.fields.reserve(fields.len());
53        for f in fields {
54            if let Expression::Wildcard {
55                schema,
56                relation,
57                is_left_joined,
58            } = &f.kind
59            {
60                let resolved_fields = resolver
61                    .list_fields(schema.as_deref(), relation)
62                    .map_err(|e| Error::ResolverFailure { inner: e })?;
63                for f in resolved_fields {
64                    self.fields.push(SelectField {
65                        ident: None,
66                        kind: Expression::Field {
67                            schema: schema.clone(),
68                            query_source: relation.clone(),
69                            field_name: f.name().to_owned(),
70                            via_left_join: *is_left_joined,
71                        },
72                    });
73                }
74            } else {
75                self.fields.push(f);
76            }
77        }
78        Ok(())
79    }
80}

82/// Infer information about a given view definition
83///
84/// This method accepts both `CREATE VIEW xyz AS SELECT …` and
85/// plain `SELECT …` statements as view definition.
86pub fn parse_view_def(definition: &str) -> Result<ViewData> {
87    let dialect = sqlparser::dialect::SQLiteDialect {};
88    let options = ParserOptions::new();
89
90    let stmt = sqlparser::parser::Parser::new(&dialect)
91        .with_options(options)
92        .try_with_sql(definition)?
93        .parse_statement()?;
94
95    let select = match stmt {
96        sqlparser::ast::Statement::Query(query) => query,
97        sqlparser::ast::Statement::CreateView { query, .. } => query,
98        stmt => {
99            return Err(Error::UnsupportedSql {
100                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unexpected statement: `{0}`",
                stmt))
    })format!("Unexpected statement: `{stmt}`"),
101            });
102        }
103    };
104    let result = match &*select.body {
105        sqlparser::ast::SetExpr::Select(select) => infer_from_select(select),
106        // we likely want to support more complex queries here as well (UNION, CTE, etc)
107        s => {
108            return Err(Error::UnsupportedSql {
109                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported query kind: `{0}`", s))
    })format!("Unsupported query kind: `{s}`"),
110            });
111        }
112    };
113    Ok(ViewData { fields: result? })
114}
