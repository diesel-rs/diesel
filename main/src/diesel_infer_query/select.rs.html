37pub enum Expression {
38    /// A literal value like `1` or `'foo'`
39    Literal {
40        /// The actual value
41        value: Option<String>,
42        /// is this value `NULL`?
43        is_null: bool,
44    },
45    /// A field of a query source
46    Field {
47        /// the schema of the query source
48        schema: Option<String>,
49        /// the name of the query source
50        query_source: String,
51        /// the name of the field
52        field_name: String,
53        /// is this field coming from a query source joined via a `LEFT JOIN`
54        via_left_join: bool,
55    },
56    /// A cast expression
57    Cast {
58        /// inner expression
59        inner: Box<Expression>,
60        /// target cast type
61        tpe: String,
62    },
63    /// A binary operation
64    BinaryOp {
65        /// left side of the operation
66        left: Box<Expression>,
67        /// right side of the operation
68        right: Box<Expression>,
69        /// operator
70        op: String,
71        /// is this operation statically known not to produce null values
72        statically_not_null: bool,
73    },
74    // A postfix operation like `IS NULL`
75    PostfixOp {
76        expr: Box<Expression>,
77        op: String,
78        statically_not_null: bool,
79    },
80    /// A function call
81    Function {
82        name: String,
83        schema: Option<String>,
84        arguments: Vec<Expression>,
85    },
86    /// A wild card `*` expression
87    Wildcard {
88        schema: Option<String>,
89        relation: String,
90        is_left_joined: bool,
91    },
92    /// A `left BETWEEN low AND high` expression
93    Between {
94        left: Box<Expression>,
95        negated: bool,
96        low: Box<Expression>,
97        high: Box<Expression>,
98    },
99    // A `CASE [operand] [conditions] ELSE [else]` expression
100    Case {
101        operand: Option<Box<Expression>>,
102        conditions: Vec<CaseCondition>,
103        else_clause: Option<Box<Expression>>,
104    },
105    /// A unknown select expression
106    Unknown,
107}

109impl Expression {
110    pub(crate) fn infer_nullability(
111        &self,
112        resolver: &mut (dyn SchemaResolver + '_),
113    ) -> Result<Option<bool>> {
114        match self {
115            Self::Literal { is_null, .. } => Ok(Some(*is_null)),
116            Self::Field {
117                via_left_join: true,
118                ..
119            } => Ok(Some(true)),
120            Self::Field {
121                schema,
122                query_source: table,
123                field_name,
124                ..
125            } => Ok(Some(
126                resolver
127                    .resolve_field(schema.as_deref(), table, field_name)
128                    .map_err(|inner| Error::ResolverFailure { inner })?
129                    .is_nullable(),
130            )),
131            Self::Cast { inner, .. } => inner.infer_nullability(resolver),
132            Self::BinaryOp {
133                left,
134                right,
135                statically_not_null,
136                ..
137            } => {
138                if *statically_not_null {
139                    Ok(Some(false))
140                } else {
141                    match (
142                        left.infer_nullability(resolver)?,
143                        right.infer_nullability(resolver)?,
144                    ) {
145                        (None, _) | (_, None) => Ok(None),
146                        (Some(a), Some(b)) => Ok(Some(a || b)),
147                    }
148                }
149            }
150            Expression::PostfixOp {
151                expr,
152                statically_not_null,
153                ..
154            } => {
155                if *statically_not_null {
156                    Ok(Some(false))
157                } else {
158                    expr.infer_nullability(resolver)
159                }
160            }
161            Expression::Function { name, schema, .. } => {
162                match name.to_lowercase().as_str() {
163                    // we consider count as only not nullable function for now
164                    "count" if schema.is_none() => Ok(Some(false)),
165                    _ => Ok(Some(true)),
166                }
167            }
168            Expression::Between {
169                left, low, high, ..
170            } => {
171                let nullability = [
172                    left.infer_nullability(resolver)?,
173                    low.infer_nullability(resolver)?,
174                    high.infer_nullability(resolver)?,
175                ];
176                Ok(nullability
177                    .into_iter()
178                    .try_fold(false, |agg, v| Some(agg || v?)))
179            }
180            Expression::Unknown => Ok(None),
181            Expression::Case {
182                conditions,
183                else_clause,
184                ..
185            } => conditions
186                .iter()
187                .map(|c| &c.result)
188                .chain(else_clause.iter().map(|c| &**c))
189                .map(|c| c.infer_nullability(resolver))
190                .try_fold(Some(false), |agg, v| match (agg, v?) {
191                    (Some(agg), Some(v)) => Ok(Some(agg || v)),
192                    _ => Ok(None),
193                }),
194            Expression::Wildcard { .. } => ::core::panicking::panic("internal error: entered unreachable code")unreachable!(),
195        }
196    }
197}

199pub(crate) fn infer_from_select(select: &sqlparser::ast::Select) -> Result<Vec<SelectField>> {
200    let query_source_lookup = collect_query_sources(&select.from)?;
201
202    select
203        .projection
204        .iter()
205        .map(|p| infer_projection(p, &query_source_lookup))
206        .collect()
207}

209pub(crate) fn collect_query_sources<'a>(
210    sources: &'a [sqlparser::ast::TableWithJoins],
211) -> Result<HashMap<&'a str, QuerySource<'a>>> {
212    let mut out = HashMap::with_capacity(sources.len());
213    for s in sources {
214        QuerySource::fill_from_table_with_joins(&mut out, s)?;
215    }
216    Ok(out)
217}

219pub(crate) fn infer_projection(
220    item: &SelectItem,
221    query_source_lookup: &HashMap<&str, QuerySource>,
222) -> Result<SelectField> {
223    match item {
224        SelectItem::UnnamedExpr(expr) => Ok(SelectField {
225            ident: None,
226            kind: crate::expression::infer_expr(expr, query_source_lookup)?,
227        }),
228        SelectItem::ExprWithAlias { expr, alias } => Ok(SelectField {
229            ident: Some(alias.value.clone()),
230            kind: crate::expression::infer_expr(expr, query_source_lookup)?,
231        }),
232        SelectItem::QualifiedWildcard(
233            SelectItemQualifiedWildcardKind::ObjectName(name),
234            _wildcard_additional_options,
235        ) => {
236            if let Some(item) = name
237                .0
238                .last()
239                .and_then(|a| a.as_ident())
240                .map(|a| a.value.as_str())
241                .and_then(|k| query_source_lookup.get(k))
242            {
243                let is_left_joined = item.contains_left_join(query_source_lookup)?;
244                Ok(SelectField {
245                    ident: None,
246                    kind: Expression::Wildcard {
247                        schema: item.schema.map(|s| s.to_owned()),
248                        relation: item.name.to_owned(),
249                        is_left_joined,
250                    },
251                })
252            } else {
253                ::core::panicking::panic("not yet implemented")todo!()
254            }
255        }
256        SelectItem::Wildcard(_) if query_source_lookup.len() == 1 => {
257            let wildcard = query_source_lookup.values().next().expect("Is exactly one");
258            Ok(SelectField {
259                ident: None,
260                kind: Expression::Wildcard {
261                    schema: wildcard.schema.map(|c| c.to_owned()),
262                    relation: wildcard.name.to_owned(),
263                    is_left_joined: false,
264                },
265            })
266        }
267        s @ SelectItem::Wildcard(_wildcard_additional_options) => Err(Error::UnsupportedSql {
268            msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported `SELECT` expression: `{0}`",
                s))
    })format!("Unsupported `SELECT` expression: `{s}`"),
269        }),
270        s @ SelectItem::QualifiedWildcard(
271            _select_item_qualified_wildcard_kind,
272            _wildcard_additional_options,
273        ) => Err(Error::UnsupportedSql {
274            msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported `SELECT` expression: `{0}`",
                s))
    })format!("Unsupported `SELECT` expression: `{s}`"),
275        }),
276    }
