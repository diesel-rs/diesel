30pub enum SelectFieldKind {
31    /// A literal value like `1` or `'foo'`
32    Literal {
33        /// The actual value
34        value: Option<String>,
35        /// is this value `NULL`?
36        is_null: bool,
37    },
38    /// A field of a query source
39    Field {
40        /// the schema of the query source
41        schema: Option<String>,
42        /// the name of the query source
43        query_source: String,
44        /// the name of the field
45        field_name: String,
46        /// is this field coming from a query source joined via a `LEFT JOIN`
47        via_left_join: bool,
48    },
49    /// A cast expression
50    Cast {
51        /// inner expression
52        inner: Box<SelectFieldKind>,
53        /// target cast type
54        tpe: String,
55    },
56    /// A unknown select expression
57    Unknown,
58}
59
60impl SelectFieldKind {
61    pub(crate) fn infer_nullability(
62        &self,
63        resolver: &mut (dyn SchemaResolver + '_),
64    ) -> Result<Option<bool>> {
65        match self {
66            // literals are null for literal `NULL` values
67            // Otherwise they cannot be null
68            SelectFieldKind::Literal { is_null, .. } => Ok(Some(*is_null)),
69            // If the field comes in via an `LEFT JOIN`
70            // we unconditionally assume it is nullable
71            SelectFieldKind::Field {
72                via_left_join: true,
73                ..
74            } => Ok(Some(true)),
75            // Otherwise we ask the schema resolver for  more information about
76            // this particular field
77            SelectFieldKind::Field {
78                schema,
79                query_source: table,
80                field_name,
81                ..
82            } => Ok(Some(
83                resolver
84                    .resolve_field(schema.as_deref(), table, field_name)
85                    .map_err(Error::ResolverFailure)?
86                    .is_nullable(),
87            )),
88            // for casts we assume that they return `NULL` only if the inner expression is `NULL`
89            SelectFieldKind::Cast { inner, .. } => inner.infer_nullability(resolver),
90            // unknown is not known, so return None
91            SelectFieldKind::Unknown => Ok(None),
92        }
93    }
94}

96pub(crate) fn infer_from_select(select: &sqlparser::ast::Select) -> Result<Vec<SelectField>> {
97    let query_source_lookup = collect_query_sources(&select.from)?;
98
99    select
100        .projection
101        .iter()
102        .map(|p| infer_projection(p, &query_source_lookup))
103        .collect()
104}

106pub(crate) fn collect_query_sources<'a>(
107    sources: &'a [sqlparser::ast::TableWithJoins],
108) -> Result<HashMap<&'a str, QuerySource<'a>>> {
109    let mut out = HashMap::with_capacity(sources.len());
110    for s in sources {
111        QuerySource::fill_from_table_with_joins(&mut out, s)?;
112    }
113    Ok(out)
114}

116pub(crate) fn infer_projection(
117    item: &SelectItem,
118    query_source_lookup: &HashMap<&str, QuerySource>,
119) -> Result<SelectField> {
120    match item {
121        SelectItem::UnnamedExpr(expr) => Ok(SelectField {
122            ident: None,
123            kind: crate::expression::infer_expr(expr, query_source_lookup)?,
124        }),
125        SelectItem::ExprWithAlias { expr, alias } => Ok(SelectField {
126            ident: Some(alias.value.clone()),
127            kind: crate::expression::infer_expr(expr, query_source_lookup)?,
128        }),
129        s @ SelectItem::QualifiedWildcard(
130            _select_item_qualified_wildcard_kind,
131            _wildcard_additional_options,
132        ) => Err(Error::UnsupportedSql {
133            msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported `SELECT` expression: `{0}`",
                s))
    })format!("Unsupported `SELECT` expression: `{s}`"),
134        }),
135        s @ SelectItem::Wildcard(_wildcard_additional_options) => Err(Error::UnsupportedSql {
136            msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported `SELECT` expression: `{0}`",
                s))
    })format!("Unsupported `SELECT` expression: `{s}`"),
137        }),
138    }
