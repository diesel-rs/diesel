54pub(crate) struct QuerySource<'a> {
55    /// The schema of the query source
56    pub(crate) schema: Option<&'a str>,
57    /// The name of the query source
58    pub(crate) name: &'a str,
59    /// The alias that is used to refer to this query source in the query
60    #[expect(dead_code, reason = "its there for later")]
61    pub(crate) alias: Option<&'a str>,
62    /// A possible join that was used to combine this query source with others
63    pub(crate) join: Option<Join>,
64}
65
66impl<'a> QuerySource<'a> {
67    /// Construct a query source from the given parser statement
68    ///
69    /// The `out` map contains a lookup list indexed by the name of the query source
70    /// as used in the query
71    pub(crate) fn fill_from_table_with_joins(
72        out: &mut HashMap<&'a str, QuerySource<'a>>,
73        table_with_joins: &'a sqlparser::ast::TableWithJoins,
74    ) -> Result<()> {
75        // first resolve the table itself
76        Self::fill_from_table_factor(&table_with_joins.relation, out)?;
77        // then any joins to this table
78        for join in &table_with_joins.joins {
79            Self::fill_from_join(out, join)?;
80        }
81        Ok(())
82    }
83
84    /// Check if this specific query source or any query source in the join chain
85    /// used to include this query source is joined via a `LEFT JOIN`
86    pub(crate) fn contains_left_join(
87        &self,
88        query_source_lookup: &HashMap<&str, QuerySource<'_>>,
89    ) -> Result<bool> {
90        if let Some(join) = &self.join {
91            match join.kind {
92                JoinKind::Inner => {
93                    let source = query_source_lookup.get(join.to.as_str()).ok_or_else(|| {
94                        Error::InvalidQuerySource {
95                            query_source: join.to.clone(),
96                        }
97                    })?;
98                    source.contains_left_join(query_source_lookup)
99                }
100                JoinKind::Left => Ok(true),
101            }
102        } else {
103            Ok(false)
104        }
105    }
106
107    fn fill_from_table_factor(
108        s: &'a sqlparser::ast::TableFactor,
109        out: &mut HashMap<&'a str, QuerySource<'a>>,
110    ) -> Result<()> {
111        match s {
112            sqlparser::ast::TableFactor::Table {
113                name,
114                alias,
115                args: None,
116                with_hints,
117                version: None,
118                with_ordinality: false,
119                partitions,
120                json_path: None,
121                sample: None,
122                index_hints,
123            }
124            // we try to be conservative with what we support here
125            // so disallow almost everything for now
126            if with_hints.is_empty()
127                && partitions.is_empty()
128                && index_hints.is_empty()
129                && name.0.len() <= 2
130                && name
131                    .0
132                    .iter()
133                    .all(|n| #[allow(non_exhaustive_omitted_patterns)] match n {
    ObjectNamePart::Identifier(_) => true,
    _ => false,
}matches!(n, ObjectNamePart::Identifier(_))) =>
134            {
135                // for a plain query source we need to extract the name (including schema if exists)
136                // and the alias for later usage
137                let (name, schema) = match name.0.as_slice() {
138                    // for a single part identifier we only have the table name
139                    [ObjectNamePart::Identifier(name)] => (&name.value, None),
140                    // for a two part identifier we also have the schema
141                    [
142                        ObjectNamePart::Identifier(schema),
143                        ObjectNamePart::Identifier(name),
144                    ] => (&name.value, Some(schema.value.as_str())),
145                    // for now anythinge else is not supported, but that should be fine
146                    _ => {
147                        return Err(Error::UnsupportedSql {
148                            msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported name: `{0}`", name))
    })format!("Unsupported name: `{name}`"),
149                        });
150                    }
151                };
152
153                let alias = alias.as_ref().map(|a| a.name.value.as_str());
154                 // if we have an alias the query source will be always referred in our query by the name of the alias
155                // So use the name of the alias in that case, otherwise the name of the table.
156                let lookup = alias.unwrap_or(name);
157                out.insert(
158                    lookup,
159                    QuerySource {
160                        schema,
161                        name,
162                        alias,
163                        join: None,
164                    },
165                );
166                Ok(())
167            }
168            sqlparser::ast::TableFactor::NestedJoin {
169                table_with_joins,
170                alias: None, // don't support aliases for joins yet
171            } => {
172                Self::fill_from_table_with_joins(out, table_with_joins)?;
173                Ok(())
174            }
175
176            s => Err(Error::UnsupportedSql {
177                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported query source: `{0}`",
                s))
    })format!("Unsupported query source: `{s}`"),
178            }),
179        }
180    }
181
182    fn fill_from_join(
183        out: &mut HashMap<&'a str, QuerySource<'a>>,
184        join: &'a sqlparser::ast::Join,
185    ) -> Result<()> {
186        use sqlparser::ast::Visit;
187
188        // get the inner parts
189        let mut inner = HashMap::new();
190        Self::fill_from_table_factor(&join.relation, &mut inner)?;
191
192        // verify that the "inner parts" is in fact one part
193        //
194        // If we ever get more than one part here we need to come up with
195        // a better solution to match the joins to the query sources
196        if inner.len() != 1 {
197            return Err(crate::error::Error::UnsupportedSql {
198                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported join clause: `{0}`",
                join))
    })format!("Unsupported join clause: `{join}`"),
199            });
200        }
201
202        // collect query source names used in the join expressions
203        //
204        // That's likely not 100% correct but a good first approximation
205        let mut collector = IdentifierCollector::default();
206        let kind = match &join.join_operator {
207            // these are equivalent
208            sqlparser::ast::JoinOperator::LeftOuter(l) | sqlparser::ast::JoinOperator::Left(l) => {
209                let _ = l.visit(&mut collector);
210                JoinKind::Left
211            }
212            // postgres rewrites `INNER JOIN` to just `JOIN`
213            sqlparser::ast::JoinOperator::Inner(i) | sqlparser::ast::JoinOperator::Join(i) => {
214                let _ = i.visit(&mut collector);
215                JoinKind::Inner
216            }
217            // Anything else is not supported for now
218            // maybe add support for more join kinds if they turn up later
219            _d => {
220                return Err(crate::error::Error::UnsupportedSql {
221                    msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported join type: `{0}`",
                join))
    })format!("Unsupported join type: `{join}`"),
222                });
223            }
224        };
225        let mut join_expr = collector.idents;
226        // remove the query source used in the join directly
227        // So for `INNER JOIN posts ON posts.user_id = users.id`
228        // remove `posts`
229        for inner in inner.keys() {
230            join_expr.remove(*inner);
231        }
232        // we should now have only one table left
233        // that's the table we are joining to
234        let to = if join_expr.len() == 1 {
235            join_expr.into_iter().next().unwrap()
236        } else {
237            // otherwise this is a complicated join
238            // we don't support yet
239            return Err(crate::error::Error::UnsupportedSql {
240                msg: ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("Unsupported join clause: `{0}`",
                join))
    })format!("Unsupported join clause: `{join}`"),
241            });
242        };
243        for (l, mut v) in inner {
244            v.join = Some(Join {
245                to: to.clone(),
246                kind,
247            });
248            out.insert(l, v);
249        }
250        Ok(())
251    }
252}

255pub(crate) struct IdentifierCollector {
256    pub(crate) idents: HashSet<String>,
257}
258
259impl sqlparser::ast::Visitor for IdentifierCollector {
260    type Break = ();
261
262    fn pre_visit_expr(&mut self, expr: &sqlparser::ast::Expr) -> ControlFlow<Self::Break> {
263        // we are just interested in identifiers
264        if let sqlparser::ast::Expr::CompoundIdentifier(i) = expr
265            && let [tbl, _] = i.as_slice()
266        {
267            self.idents.insert(tbl.value.clone());
268        }
269        ControlFlow::Continue(())
270    }
