7#[non_exhaustive]
8pub enum Error {
9    /// Parsing the SQL failed with the provided error message
10    #[error("Failed to parse sql: {0:?}")]
11    #[non_exhaustive]
12    ParserError(#[from] sqlparser::parser::ParserError),
13    /// Handling this kind of SQL is currently not supported by this crate
14    #[error("Unsupported SQL: {msg}")]
15    UnsupportedSql {
16        /// details about the unsupported SQL expression
17        msg: String,
18    },
19    /// The query referenced a unknown query source
20    #[error("Querysource was not found in the from clause: `{query_source}`")]
21    #[non_exhaustive]
22    InvalidQuerySource {
23        /// Which query source is unknown
24        query_source: String,
25    },
26    /// The schema resolver returned an error
27    #[error("Could not resolve view data: {inner}")]
28    #[non_exhaustive]
29    ResolverFailure {
30        /// The inner resolver failure
31        inner: Box<dyn std::error::Error + Send + Sync + 'static>,
32    },
33}
