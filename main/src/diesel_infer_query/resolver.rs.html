5/// A generic interface that allows this crate
6/// to request more information about certain database
7/// relations
8pub trait SchemaResolver {
9    /// Resolve a specific database field
10    fn resolve_field<'s>(
11        &'s mut self,
12        relation_schema: Option<&str>,
13        query_relation: &str,
14        field_name: &str,
15    ) -> Result<&'s dyn SchemaField, Box<dyn std::error::Error + Send + Sync + 'static>>;
16
17    /// Get a list of fields for a specific database relation
18    /// in the order returned by the database
19    fn list_fields<'s>(
20        &'s mut self,
21        relation_schema: Option<&str>,
22        query_relation: &str,
23    ) -> Result<Vec<&'s dyn SchemaField>, Box<dyn std::error::Error + Send + Sync + 'static>>;
24}
25
26/// A generic representation of a database field
27pub trait SchemaField {
28    /// Is this field nullable
29    fn is_nullable(&self) -> bool;
30    /// The database side name of the field
31    fn name(&self) -> &str;
32}
