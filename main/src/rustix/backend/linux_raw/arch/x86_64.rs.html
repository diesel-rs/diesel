11#[inline]
12pub(in crate::backend) unsafe fn syscall0_readonly(nr: SyscallNumber<'_>) -> RetReg<R0> {
13    let r0;
14    asm!(
15        "syscall",
16        inlateout("rax") nr.to_asm() => r0,
17        lateout("rcx") _,
18        lateout("r11") _,
19        options(nostack, preserves_flags, readonly)
20    );
21    FromAsm::from_asm(r0)
22}
