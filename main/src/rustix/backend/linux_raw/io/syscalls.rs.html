34#[inline]
35pub(crate) unsafe fn read(fd: BorrowedFd<'_>, buf: (*mut u8, usize)) -> io::Result<usize> {
36    let r = ret_usize(crate::backend::arch::choose::syscall3(crate::backend::reg::nr(linux_raw_sys::general::__NR_read),
    fd.into(), buf.0.into(), pass_usize(buf.1).into())syscall!(__NR_read, fd, buf.0, pass_usize(buf.1)));
37
38    #[cfg(sanitize_memory)]
39    if let Ok(len) = r {
40        crate::msan::unpoison(buf.0, len);
41    }
42
43    r
44}

46#[inline]
47pub(crate) unsafe fn pread(
48    fd: BorrowedFd<'_>,
49    buf: (*mut u8, usize),
50    pos: u64,
51) -> io::Result<usize> {
52    // <https://git.kernel.org/pub/scm/linux/kernel/git/torvalds/linux.git/tree/arch/arm64/kernel/sys32.c?h=v6.13#n70>
53    #[cfg(all(
54        target_pointer_width = "32",
55        any(
56            target_arch = "arm",
57            target_arch = "mips",
58            target_arch = "mips32r6",
59            target_arch = "powerpc"
60        ),
61    ))]
62    let r = ret_usize(syscall!(
63        __NR_pread64,
64        fd,
65        buf.0,
66        pass_usize(buf.1),
67        zero(),
68        hi(pos),
69        lo(pos)
70    ));
71
72    #[cfg(all(
73        target_pointer_width = "32",
74        not(any(
75            target_arch = "arm",
76            target_arch = "mips",
77            target_arch = "mips32r6",
78            target_arch = "powerpc"
79        )),
80    ))]
81    let r = ret_usize(syscall!(
82        __NR_pread64,
83        fd,
84        buf.0,
85        pass_usize(buf.1),
86        hi(pos),
87        lo(pos)
88    ));
89
90    #[cfg(target_pointer_width = "64")]
91    let r = ret_usize(crate::backend::arch::choose::syscall4(crate::backend::reg::nr(linux_raw_sys::general::__NR_pread64),
    fd.into(), buf.0.into(), pass_usize(buf.1).into(),
    loff_t_from_u64(pos).into())syscall!(
92        __NR_pread64,
93        fd,
94        buf.0,
95        pass_usize(buf.1),
96        loff_t_from_u64(pos)
97    ));
98
99    #[cfg(sanitize_memory)]
100    if let Ok(len) = r {
101        crate::msan::unpoison(buf.0, len);
102    }
103
104    r
105}

108pub(crate) fn readv(fd: BorrowedFd<'_>, bufs: &mut [IoSliceMut<'_>]) -> io::Result<usize> {
109    let (bufs_addr, bufs_len) = slice(&bufs[..cmp::min(bufs.len(), MAX_IOV)]);
110
111    unsafe { ret_usize(crate::backend::arch::choose::syscall3(crate::backend::reg::nr(linux_raw_sys::general::__NR_readv),
    fd.into(), bufs_addr.into(), bufs_len.into())syscall!(__NR_readv, fd, bufs_addr, bufs_len)) }
112}

114#[inline]
115pub(crate) fn preadv(
116    fd: BorrowedFd<'_>,
117    bufs: &mut [IoSliceMut<'_>],
118    pos: u64,
119) -> io::Result<usize> {
120    let (bufs_addr, bufs_len) = slice(&bufs[..cmp::min(bufs.len(), MAX_IOV)]);
121
122    // Unlike the plain "p" functions, the "pv" functions pass their offset in
123    // an endian-independent way, and always in two registers.
124    unsafe {
125        ret_usize(crate::backend::arch::choose::syscall5(crate::backend::reg::nr(linux_raw_sys::general::__NR_preadv),
    fd.into(), bufs_addr.into(), bufs_len.into(),
    pass_usize(pos as usize).into(), pass_usize((pos >> 32) as usize).into())syscall!(
126            __NR_preadv,
127            fd,
128            bufs_addr,
129            bufs_len,
130            pass_usize(pos as usize),
131            pass_usize((pos >> 32) as usize)
132        ))
133    }
134}

273pub(crate) unsafe fn close(fd: RawFd) {
274    // See the documentation for [`io::close`] for why errors are ignored.
275    crate::backend::arch::choose::syscall1_readonly(crate::backend::reg::nr(linux_raw_sys::general::__NR_close),
    raw_fd(fd).into())syscall_readonly!(__NR_close, raw_fd(fd)).decode_void();
276}

307#[allow(clippy::needless_pass_by_ref_mut)]
308#[inline]
309pub(crate) fn dup2(fd: BorrowedFd<'_>, new: &mut OwnedFd) -> io::Result<()> {
310    #[cfg(any(target_arch = "aarch64", target_arch = "riscv64"))]
311    {
312        // We don't need to worry about the difference between `dup2` and
313        // `dup3` when the file descriptors are equal because we have an
314        // `&mut OwnedFd` which means `fd` doesn't alias it.
315        dup3(fd, new, DupFlags::empty())
316    }
317
318    #[cfg(not(any(target_arch = "aarch64", target_arch = "riscv64")))]
319    unsafe {
320        ret_discarded_fd(crate::backend::arch::choose::syscall2_readonly(crate::backend::reg::nr(linux_raw_sys::general::__NR_dup2),
    fd.into(), new.as_fd().into())syscall_readonly!(__NR_dup2, fd, new.as_fd()))
321    }
322}

330#[inline]
331pub(crate) fn fcntl_getfd(fd: BorrowedFd<'_>) -> io::Result<FdFlags> {
332    #[cfg(target_pointer_width = "32")]
333    unsafe {
334        ret_c_uint(syscall_readonly!(__NR_fcntl64, fd, c_uint(F_GETFD)))
335            .map(FdFlags::from_bits_retain)
336    }
337    #[cfg(target_pointer_width = "64")]
338    unsafe {
339        ret_c_uint(crate::backend::arch::choose::syscall2_readonly(crate::backend::reg::nr(linux_raw_sys::general::__NR_fcntl),
    fd.into(), c_uint(F_GETFD).into())syscall_readonly!(__NR_fcntl, fd, c_uint(F_GETFD)))
340            .map(FdFlags::from_bits_retain)
341    }
342}
