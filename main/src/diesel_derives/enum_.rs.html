142            return Err(syn::Error::new(input.span(), ERROR_MESSAGE));

147                return Err(syn::Error::new(input.span(), ERROR_MESSAGE));

151            &input.attrs,

164            if !v.fields.is_empty() {
165                return Err(syn::Error::new(v.span(), ERROR_MESSAGE));
166            }
