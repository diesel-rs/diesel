62    impl ParseDeprecated for StructAttr {
63        fn parse_deprecated(input: ParseStream) -> Result<Option<Self>> {
64            let name: Ident = input.parse()?;
65            let name_str = name.to_string();
66
67            match &*name_str {
68                "table_name" => {
69                    let lit_str = parse_eq_and_lit_str(name.clone(), input, TABLE_NAME_NOTE)?;
70                    warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("use `#[diesel(table_name = {0})]` instead",
                        lit_str.value()))
            }));warn!(
71                        name,
72                        &format!("use `#[diesel(table_name = {})]` instead", lit_str.value())
73                    );
74                    Ok(Some(StructAttr::TableName(name, lit_str.parse()?)))
75                }
76                "changeset_options" => {
77                    let (ident, value) = parse_changeset_options(name.clone(), input)?;
78                    warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("use `#[diesel(treat_none_as_null = {0})]` instead",
                        value.value))
            }));warn!(
79                        name,
80                        &format!(
81                            "use `#[diesel(treat_none_as_null = {})]` instead",
82                            value.value
83                        )
84                    );
85                    Ok(Some(StructAttr::TreatNoneAsNull(ident, value)))
86                }
87                "sql_type" => {
88                    let lit_str = parse_eq_and_lit_str(name.clone(), input, SQL_TYPE_NOTE)?;
89                    warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("use `#[diesel(sql_type = {0})]` instead",
                        lit_str.value()))
            }));warn!(
90                        name,
91                        &format!("use `#[diesel(sql_type = {})]` instead", lit_str.value())
92                    );
93                    Ok(Some(StructAttr::SqlType(name, lit_str.parse()?)))
94                }
95                "primary_key" => {
96                    let keys = parse_primary_key(name.clone(), input)?;
97                    let hint = keys
98                        .iter()
99                        .map(|i| i.to_string())
100                        .collect::<Vec<_>>()
101                        .join(", ");
102                    warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("use `#[diesel(primary_key({0}))]` instead",
                        hint))
            }));warn!(
103                        name,
104                        &format!("use `#[diesel(primary_key({hint}))]` instead")
105                    );
106                    Ok(Some(StructAttr::PrimaryKey(name, keys)))
107                }
108                "belongs_to" => {
109                    let belongs_to = parse_belongs_to(name.clone(), input)?;
110                    let parent = belongs_to
111                        .parent
112                        .path
113                        .segments
114                        .iter()
115                        .map(|s| s.ident.to_string())
116                        .collect::<Vec<_>>()
117                        .join("::");
118                    if let Some(ref key) = belongs_to.foreign_key {
119                        warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("use `#[diesel(belongs_to({0}, foreign_key = {1}))]` instead",
                        parent, key))
            }));warn!(
120                            name,
121                            &format!(
122                                "use `#[diesel(belongs_to({parent}, foreign_key = {key}))]` instead"
123                            )
124                        );
125                    } else {
126                        warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("use `#[diesel(belongs_to({0}))]` instead",
                        parent))
            }));warn!(
127                            name,
128                            &format!("use `#[diesel(belongs_to({parent}))]` instead")
129                        );
130                    }
131                    Ok(Some(StructAttr::BelongsTo(name, belongs_to)))
132                }
133                "sqlite_type" => {
134                    let name_value = parse_eq_and_lit_str(name.clone(), input, SQLITE_TYPE_NOTE)?;
135                    warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("use `#[diesel(sqlite_type(name = \"{0}\"))]` instead",
                        name_value.value()))
            }));warn!(
136                        name,
137                        &format!(
138                            "use `#[diesel(sqlite_type(name = \"{}\"))]` instead",
139                            name_value.value()
140                        )
141                    );
142                    Ok(Some(StructAttr::SqliteType(
143                        name,
144                        SqliteType { name: name_value },
145                    )))
146                }
147                "mysql_type" => {
148                    let name_value = parse_eq_and_lit_str(name.clone(), input, MYSQL_TYPE_NOTE)?;
149                    warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("use `#[diesel(mysql_type(name = \"{0}\"))]` instead",
                        name_value.value()))
            }));warn!(
150                        name,
151                        &format!(
152                            "use `#[diesel(mysql_type(name = \"{}\"))]` instead",
153                            name_value.value()
154                        )
155                    );
156                    Ok(Some(StructAttr::MysqlType(
157                        name,
158                        MysqlType { name: name_value },
159                    )))
160                }
161                "postgres" => {
162                    let pg_type = parse_postgres_type(name.clone(), input)?;
163                    let msg = match &pg_type {
164                        PostgresType::Fixed(oid, array_oid) => ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("use `#[diesel(postgres_type(oid = {0}, array_oid = {1}))]` instead",
                oid.base10_parse::<u32>()?, array_oid.base10_parse::<u32>()?))
    })format!(
165                            "use `#[diesel(postgres_type(oid = {}, array_oid = {}))]` instead",
166                            oid.base10_parse::<u32>()?,
167                            array_oid.base10_parse::<u32>()?
168                        ),
169                        PostgresType::Lookup(name, Some(schema)) => ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("use `#[diesel(postgres_type(name = \"{0}\", schema = \"{1}\"))]` instead",
                name.value(), schema.value()))
    })format!(
170                            "use `#[diesel(postgres_type(name = \"{}\", schema = \"{}\"))]` instead",
171                            name.value(),
172                            schema.value()
173                        ),
174                        PostgresType::Lookup(name, None) => ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("use `#[diesel(postgres_type(name = \"{0}\"))]` instead",
                name.value()))
    })format!(
175                            "use `#[diesel(postgres_type(name = \"{}\"))]` instead",
176                            name.value(),
177                        ),
178                    };
179
180                    warn(name.span(),
    &::alloc::__export::must_use({
                ::alloc::fmt::format(format_args!("#[{0}] attribute form is deprecated",
                        name))
            }), &msg);warn!(name, &msg);
181                    Ok(Some(StructAttr::PostgresType(name, pg_type)))
182                }
183                _ => Ok(None),
184            }
185        }
186    }
