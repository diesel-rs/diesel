74impl fmt::Display for StageParamsObject {
75    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
76        let url = &self.url.as_ref();
77        let storage_integration = &self.storage_integration.as_ref();
78        let endpoint = &self.endpoint.as_ref();
79
80        if url.is_some() {
81            f.write_fmt(format_args!(" URL=\'{0}\'", url.unwrap()))write!(f, " URL='{}'", url.unwrap())?;
82        }
83        if storage_integration.is_some() {
84            f.write_fmt(format_args!(" STORAGE_INTEGRATION={0}",
        storage_integration.unwrap()))write!(f, " STORAGE_INTEGRATION={}", storage_integration.unwrap())?;
85        }
86        if endpoint.is_some() {
87            f.write_fmt(format_args!(" ENDPOINT=\'{0}\'", endpoint.unwrap()))write!(f, " ENDPOINT='{}'", endpoint.unwrap())?;
88        }
89        if !self.credentials.options.is_empty() {
90            f.write_fmt(format_args!(" CREDENTIALS=({0})", self.credentials))write!(f, " CREDENTIALS=({})", self.credentials)?;
91        }
92        if !self.encryption.options.is_empty() {
93            f.write_fmt(format_args!(" ENCRYPTION=({0})", self.encryption))write!(f, " ENCRYPTION=({})", self.encryption)?;
94        }
95
96        Ok(())
97    }
98}
