14pub fn create_named(
15    path: &Path,
16    open_options: &mut OpenOptions,
17    #[cfg_attr(target_os = "wasi", allow(unused))] permissions: Option<&std::fs::Permissions>,
18) -> io::Result<File> {
19    open_options.read(true).write(true).create_new(true);
20
21    #[cfg(not(target_os = "wasi"))]
22    {
23        use std::os::unix::fs::{OpenOptionsExt, PermissionsExt};
24        open_options.mode(permissions.map(|p| p.mode()).unwrap_or(0o600));
25    }
26
27    open_options.open(path)
28}

30fn create_unlinked(path: &Path) -> io::Result<File> {
31    let tmp;
32    // shadow this to decrease the lifetime. It can't live longer than `tmp`.
33    let mut path = path;
34    if !path.is_absolute() {
35        let cur_dir = std::env::current_dir()?;
36        tmp = cur_dir.join(path);
37        path = &tmp;
38    }
39
40    let f = create_named(path, &mut OpenOptions::new(), None)?;
41    // don't care whether the path has already been unlinked,
42    // but perhaps there are some IO error conditions we should send up?
43    let _ = fs::remove_file(path);
44    Ok(f)
45}

47#[cfg(target_os = "linux")]
48pub fn create(dir: &Path) -> io::Result<File> {
49    use rustix::{fs::OFlags, io::Errno};
50    use std::os::unix::fs::OpenOptionsExt;
51    OpenOptions::new()
52        .read(true)
53        .write(true)
54        .custom_flags(OFlags::TMPFILE.bits() as i32) // do not mix with `create_new(true)`
55        .open(dir)
56        .or_else(|e| {
57            match Errno::from_io_error(&e) {
58                // These are the three "not supported" error codes for O_TMPFILE.
59                Some(Errno::OPNOTSUPP) | Some(Errno::ISDIR) | Some(Errno::NOENT) => {
60                    create_unix(dir)
61                }
62                _ => Err(e),
63            }
64        })
65}

82pub fn reopen(file: &File, path: &Path) -> io::Result<File> {
83    let new_file = OpenOptions::new().read(true).write(true).open(path)?;
84    let old_meta = rustix::fs::fstat(file)?;
85    let new_meta = rustix::fs::fstat(&new_file)?;
86    if old_meta.st_dev != new_meta.st_dev || old_meta.st_ino != new_meta.st_ino {
87        return Err(io::Error::new(
88            io::ErrorKind::NotFound,
89            "original tempfile has been replaced",
90        ));
91    }
92    Ok(new_file)
93}

96pub fn persist(old_path: &Path, new_path: &Path, overwrite: bool) -> io::Result<()> {
97    if overwrite {
98        rename(old_path, new_path)?;
99    } else {
100        // On Linux and apple operating systems, use `renameat_with` to avoid overwriting an
101        // existing name, if the kernel and the filesystem support it.
102        #[cfg(any(
103            target_os = "android",
104            target_os = "linux",
105            target_os = "macos",
106            target_os = "ios",
107            target_os = "tvos",
108            target_os = "visionos",
109            target_os = "watchos",
110        ))]
111        {
112            use rustix::fs::{renameat_with, RenameFlags, CWD};
113            use rustix::io::Errno;
114            use std::sync::atomic::{AtomicBool, Ordering::Relaxed};
115
116            static NOSYS: AtomicBool = AtomicBool::new(false);
117            if !NOSYS.load(Relaxed) {
118                match renameat_with(CWD, old_path, CWD, new_path, RenameFlags::NOREPLACE) {
119                    Ok(()) => return Ok(()),
120                    Err(Errno::NOSYS) => NOSYS.store(true, Relaxed),
121                    Err(Errno::INVAL) => {}
122                    Err(e) => return Err(e.into()),
123                }
124            }
125        }
126
127        // Otherwise use `hard_link` to create the new filesystem name, which
128        // will fail if the name already exists, and then `unlink` to remove
129        // the old name.
130        hard_link(old_path, new_path)?;
131
132        // Ignore unlink errors. Can we do better?
133        let _ = unlink(old_path);
134    }
135    Ok(())
136}
