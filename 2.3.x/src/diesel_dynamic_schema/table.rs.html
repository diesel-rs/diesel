19impl<T, U> Table<T, U> {

21        Self { name, schema: None }
22    }
23
24    pub(crate) fn with_schema(schema: U, name: T) -> Self {
25        Self {
26            name,

31    /// Create a column with this table.

33    where
34        Self: Clone,
35    {

98        out.unsafe_to_cache_prepared();
99
100        if let Some(ref schema) = self.schema {
