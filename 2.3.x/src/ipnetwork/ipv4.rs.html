304/// Creates an `Ipv4Network` from parsing a string in CIDR notation.
305///
306/// # Examples
307///
308/// ```
309/// use std::net::Ipv4Addr;
310/// use ipnetwork::Ipv4Network;
311///
312/// let new = Ipv4Network::new(Ipv4Addr::new(10, 1, 9, 32), 16).unwrap();
313/// let from_cidr: Ipv4Network = "10.1.9.32/16".parse().unwrap();
314/// assert_eq!(new.ip(), from_cidr.ip());
315/// assert_eq!(new.prefix(), from_cidr.prefix());
316/// ```
317impl FromStr for Ipv4Network {
318    type Err = IpNetworkError;
319    fn from_str(s: &str) -> Result<Self, Self::Err> {
320        let (addr_str, prefix_str) = cidr_parts(s)?;
321        let addr = Ipv4Addr::from_str(addr_str)?;
322        let prefix = match prefix_str {
323            Some(v) => {
324                if let Ok(netmask) = Ipv4Addr::from_str(v) {
325                    ipv4_mask_to_prefix(netmask)?
326                } else {
327                    parse_prefix(v, IPV4_BITS)?
328                }
329            }
330            None => IPV4_BITS,
331        };
332        Ipv4Network::new(addr, prefix)
333    }
334}

381/// Converts a `Ipv4Addr` network mask into a prefix.
382///
383/// If the mask is invalid this will return an `IpNetworkError::InvalidPrefix`.
384pub fn ipv4_mask_to_prefix(mask: Ipv4Addr) -> Result<u8, IpNetworkError> {
385    let mask = u32::from(mask);
386
387    let prefix = (!mask).leading_zeros() as u8;
388    if (u64::from(mask) << prefix) & 0xffff_ffff != 0 {
389        Err(IpNetworkError::InvalidPrefix)
390    } else {
391        Ok(prefix)
392    }
393}
