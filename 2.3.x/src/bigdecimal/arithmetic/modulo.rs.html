9/// optimized calculation of n % 10
10pub(crate) fn mod_ten_2p64_le(mut digits: impl Iterator<Item = u64>) -> u8 {
11    let d0 = digits.next().unwrap_or(0) % 10;
12    let mut acc: u64 = digits.map(|d| d % 10).sum();
13    acc *= 6;
14    acc += d0;
15    (acc % 10) as u8
16}
