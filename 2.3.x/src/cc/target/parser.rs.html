16    fn from_cargo_environment_variables() -> Result<Self, Error> {
17        // `TARGET` must be present.
18        //
19        // No need to emit `rerun-if-env-changed` for this,
20        // as it is controlled by Cargo itself.
21        #[allow(clippy::disallowed_methods)]
22        let target_name = env::var("TARGET").map_err(|err| {
23            Error::new(
24                ErrorKind::EnvVarNotFound,
25                ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("failed reading TARGET: {0}", err))
    })format!("failed reading TARGET: {err}"),
26            )
27        })?;
28
29        // Parse the full architecture name from the target name.
30        let (full_arch, _rest) = target_name.split_once('-').ok_or(Error::new(
31            ErrorKind::InvalidTarget,
32            ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("target `{0}` only had a single component (at least two required)",
                target_name))
    })format!("target `{target_name}` only had a single component (at least two required)"),
33        ))?;
34
35        let cargo_env = |name, fallback: Option<&str>| -> Result<Box<str>, Error> {
36            // No need to emit `rerun-if-env-changed` for these,
37            // as they are controlled by Cargo itself.
38            #[allow(clippy::disallowed_methods)]
39            match env::var(name) {
40                Ok(var) => Ok(var.into_boxed_str()),
41                Err(err) => match fallback {
42                    Some(fallback) => Ok(fallback.into()),
43                    None => Err(Error::new(
44                        ErrorKind::EnvVarNotFound,
45                        ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("did not find fallback information for target `{0}`, and failed reading {1}: {2}",
                target_name, name, err))
    })format!("did not find fallback information for target `{target_name}`, and failed reading {name}: {err}"),
46                    )),
47                },
48            }
49        };
50
51        // Prefer to use `CARGO_ENV_*` if set, since these contain the most
52        // correct information relative to the current `rustc`, and makes it
53        // possible to support custom target JSON specs unknown to `rustc`.
54        //
55        // NOTE: If the user is using an older `rustc`, that data may be older
56        // than our pre-generated data, but we still prefer Cargo's view of
57        // the world, since at least `cc` won't differ from `rustc` in that
58        // case.
59        //
60        // These may not be set in case the user depended on being able to
61        // just set `TARGET` outside of build scripts; in those cases, fall
62        // back back to data from the known set of target names instead.
63        //
64        // See discussion in #1225 for further details.
65        let fallback_target = TargetInfo::from_rustc_target(&target_name).ok();
66        let ft = fallback_target.as_ref();
67        let arch = cargo_env("CARGO_CFG_TARGET_ARCH", ft.map(|t| t.arch))?;
68        let vendor = cargo_env("CARGO_CFG_TARGET_VENDOR", ft.map(|t| t.vendor))?;
69        let os = cargo_env("CARGO_CFG_TARGET_OS", ft.map(|t| t.os))?;
70        let mut env = cargo_env("CARGO_CFG_TARGET_ENV", ft.map(|t| t.env))?;
71        // `target_abi` was stabilized in Rust 1.78, which is higher than our
72        // MSRV, so it may not always be available; In that case, fall back to
73        // `""`, which is _probably_ correct for unknown target names.

78        // TODO: Remove once MSRV is bumped to 1.91 and `rustc` removes these from `target_abi`.
79        if #[allow(non_exhaustive_omitted_patterns)] match &*abi {
    "macabi" | "sim" => true,
    _ => false,
}matches!(&*abi, "macabi" | "sim") {
80            if true {
    if !#[allow(non_exhaustive_omitted_patterns)] match &*env {
                "" | "macbi" | "sim" => true,
                _ => false,
            } {
        {
            ::std::rt::panic_fmt(format_args!("env/abi mismatch: {0:?}, {1:?}",
                    env, abi));
        }
    };
};debug_assert!(
81                matches!(&*env, "" | "macbi" | "sim"),
82                "env/abi mismatch: {:?}, {:?}",
83                env,
84                abi,
85            );
86            env = mem::replace(&mut abi, String::default().into_boxed_str());
87        }
88
89        Ok(Self {
90            full_arch: full_arch.to_string().into_boxed_str(),

92            vendor,
93            os,
94            env,
95            abi,

130/// Parse the full architecture in the target name into the simpler
131/// `cfg(target_arch = "...")` that `rustc` exposes.
132fn parse_arch(full_arch: &str) -> Option<&str> {
133    // NOTE: Some of these don't necessarily match an existing target in
134    // `rustc`. They're parsed anyhow to be as forward-compatible as possible,
135    // while still being correct.
136    //
137    // See also:
138    // https://docs.rs/cfg-expr/0.18.0/cfg_expr/targets/index.html
139    // https://docs.rs/target-lexicon/0.13.2/target_lexicon/enum.Architecture.html
140    // https://gcc.gnu.org/onlinedocs/gcc/Submodel-Options.html
141    // `clang -print-targets`
142    Some(match full_arch {
143        arch if arch.starts_with("mipsisa32r6") => "mips32r6", // mipsisa32r6 | mipsisa32r6el
144        arch if arch.starts_with("mipsisa64r6") => "mips64r6", // mipsisa64r6 | mipsisa64r6el
145
146        arch if arch.starts_with("mips64") => "mips64", // mips64 | mips64el
147        arch if arch.starts_with("mips") => "mips",     // mips | mipsel
148
149        arch if arch.starts_with("loongarch64") => "loongarch64",
150        arch if arch.starts_with("loongarch32") => "loongarch32",
151
152        arch if arch.starts_with("powerpc64") => "powerpc64", // powerpc64 | powerpc64le
153        arch if arch.starts_with("powerpc") => "powerpc",
154        arch if arch.starts_with("ppc64") => "powerpc64",
155        arch if arch.starts_with("ppc") => "powerpc",
156
157        arch if arch.starts_with("x86_64") => "x86_64", // x86_64 | x86_64h
158        arch if arch.starts_with("i") && arch.ends_with("86") => "x86", // i386 | i586 | i686
159
160        "arm64ec" => "arm64ec", // https://github.com/rust-lang/rust/issues/131172
161        arch if arch.starts_with("aarch64") => "aarch64", // arm64e | arm64_32
162        arch if arch.starts_with("arm64") => "aarch64", // aarch64 | aarch64_be
163
164        arch if arch.starts_with("arm") => "arm", // arm | armv7s | armeb | ...
165        arch if arch.starts_with("thumb") => "arm", // thumbv4t | thumbv7a | thumbv8m | ...
166
167        arch if arch.starts_with("riscv64") => "riscv64",
168        arch if arch.starts_with("riscv32") => "riscv32",
169
170        arch if arch.starts_with("wasm64") => "wasm64",
171        arch if arch.starts_with("wasm32") => "wasm32", // wasm32 | wasm32v1
172        "asmjs" => "wasm32",
173
174        arch if arch.starts_with("nvptx64") => "nvptx64",
175        arch if arch.starts_with("nvptx") => "nvptx",
176
177        arch if arch.starts_with("bpf") => "bpf", // bpfeb | bpfel
178
179        // https://github.com/bytecodealliance/wasmtime/tree/v30.0.1/pulley
180        arch if arch.starts_with("pulley64") => "pulley64",
181        arch if arch.starts_with("pulley32") => "pulley32",
182
183        // https://github.com/Clever-ISA/Clever-ISA
184        arch if arch.starts_with("clever") => "clever",
185
186        "sparc" | "sparcv7" | "sparcv8" => "sparc",
187        "sparc64" | "sparcv9" => "sparc64",
188
189        "amdgcn" => "amdgpu",
190        "avr" => "avr",
191        "csky" => "csky",
192        "hexagon" => "hexagon",
193        "m68k" => "m68k",
194        "msp430" => "msp430",
195        "r600" => "r600",
196        "s390x" => "s390x",
197        "xtensa" => "xtensa",
198
199        // Arches supported by gcc, but not LLVM.
200        arch if arch.starts_with("alpha") => "alpha", // DEC Alpha
201        "hppa" => "hppa", // https://en.wikipedia.org/wiki/PA-RISC, also known as HPPA
202        arch if arch.starts_with("sh") => "sh", // SuperH
203        _ => return None,
204    })
205}

207/// Parse environment and ABI from the last component of the target name.
208fn parse_envabi(last_component: &str) -> Option<(&str, &str)> {
209    let (env, abi) = match last_component {
210        // Combined environment and ABI
211
212        // gnullvm | gnueabi | gnueabihf | gnuabiv2 | gnuabi64 | gnuspe | gnux32 | gnu_ilp32
213        env_and_abi if env_and_abi.starts_with("gnu") => {
214            let abi = env_and_abi.strip_prefix("gnu").unwrap();
215            let abi = abi.strip_prefix("_").unwrap_or(abi);
216            ("gnu", abi)
217        }
218        // musl | musleabi | musleabihf | muslabi64 | muslspe
219        env_and_abi if env_and_abi.starts_with("musl") => {
220            ("musl", env_and_abi.strip_prefix("musl").unwrap())
221        }
222        // uclibc | uclibceabi | uclibceabihf
223        env_and_abi if env_and_abi.starts_with("uclibc") => {
224            ("uclibc", env_and_abi.strip_prefix("uclibc").unwrap())
225        }
226        // newlib | newlibeabihf
227        env_and_abi if env_and_abi.starts_with("newlib") => {
228            ("newlib", env_and_abi.strip_prefix("newlib").unwrap())
229        }
230
231        // Environments
232        "msvc" => ("msvc", ""),
233        "ohos" => ("ohos", ""),
234        "qnx700" => ("nto70", ""),
235        "qnx710_iosock" => ("nto71_iosock", ""),
236        "qnx710" => ("nto71", ""),
237        "qnx800" => ("nto80", ""),
238        "sgx" => ("sgx", ""),
239        "threads" => ("threads", ""),
240        "mlibc" => ("mlibc", ""),
241
242        // ABIs
243        "abi64" => ("", "abi64"),
244        "abiv2" => ("", "spe"),
245        "eabi" => ("", "eabi"),
246        "eabihf" => ("", "eabihf"),
247        "macabi" => ("macabi", ""),
248        "sim" => ("sim", ""),
249        "softfloat" => ("", "softfloat"),
250        "spe" => ("", "spe"),
251        "x32" => ("", "x32"),
252
253        // Badly named targets, ELF is already known from target OS.
254        // Probably too late to fix now though.
255        "elf" => ("", ""),
256        // Undesirable to expose to user code (yet):
257        // https://github.com/rust-lang/rust/pull/131166#issuecomment-2389541917
258        "freestanding" => ("", ""),
259
260        _ => return None,
261    };
262    Some((env, abi))
263}

265impl<'a> TargetInfo<'a> {
266    pub(crate) fn from_rustc_target(target: &'a str) -> Result<Self, Error> {
267        // FIXME(madsmtm): This target should be renamed, cannot be parsed
268        // with the means we do below (since `none` must not be interpreted
269        // as an env/ABI).
270        if target == "x86_64-unknown-linux-none" {
271            return Ok(Self {
272                full_arch: "x86_64",
273                arch: "x86_64",
274                vendor: "unknown",
275                os: "linux",
276                env: "",
277                abi: "",
278            });
279        }
280
281        if target == "armv7a-vex-v5" {
282            return Ok(Self {
283                full_arch: "armv7a",
284                arch: "arm",
285                vendor: "vex",
286                os: "vexos",
287                env: "v5",
288                abi: "eabihf",
289            });
290        }
291
292        let mut components = target.split('-');
293
294        // Insist that the target name contains at least a valid architecture.
295        let full_arch = components.next().ok_or(Error::new(
296            ErrorKind::InvalidTarget,
297            "target was empty".to_string(),
298        ))?;
299        let arch = parse_arch(full_arch).ok_or_else(|| {
300            Error::new(
301                ErrorKind::UnknownTarget,
302                ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("target `{0}` had an unknown architecture",
                target))
    })format!("target `{target}` had an unknown architecture"),
303            )
304        })?;
305
306        // Newer target names have begun omitting the vendor, so the only
307        // component we know must be there is the OS name.
308        let components: Vec<_> = components.collect();
309        let (vendor, os, mut env, mut abi) = match &*components {
310            [] => {
311                return Err(Error::new(
312                    ErrorKind::InvalidTarget,
313                    ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("target `{0}` must have at least two components",
                target))
    })format!("target `{target}` must have at least two components"),
314                ))
315            }
316            // Two components; format is `arch-os`.
317            [os] => ("unknown", *os, "", ""),
318            // The three-component case is a bit tricky to handle, it could
319            // either have the format `arch-vendor-os` or `arch-os-env+abi`.
320            [vendor_or_os, os_or_envabi] => {
321                // We differentiate between these by checking if the last
322                // component is an env/ABI; if it isn't, then it's probably
323                // an OS instead.
324                if let Some((env, abi)) = parse_envabi(os_or_envabi) {
325                    ("unknown", *vendor_or_os, env, abi)
326                } else {
327                    (*vendor_or_os, *os_or_envabi, "", "")
328                }
329            }
330            // Four components; format is `arch-vendor-os-env+abi`.
331            [vendor, os, envabi] => {
332                let (env, abi) = parse_envabi(envabi).ok_or_else(|| {
333                    Error::new(
334                        ErrorKind::UnknownTarget,
335                        ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("unknown environment/ABI `{0}` in target `{1}`",
                envabi, target))
    })format!("unknown environment/ABI `{envabi}` in target `{target}`"),
336                    )
337                })?;
338                (*vendor, *os, env, abi)
339            }
340            _ => {
341                return Err(Error::new(
342                    ErrorKind::InvalidTarget,
343                    ::alloc::__export::must_use({
        ::alloc::fmt::format(format_args!("too many components in target `{0}`",
                target))
    })format!("too many components in target `{target}`"),
344                ))
345            }
346        };
347
348        // Part of the architecture name is carried over into the ABI.
349        match full_arch {
350            // https://github.com/rust-lang/compiler-team/issues/830
351            arch if arch.starts_with("riscv32e") => {
352                abi = "ilp32e";
353            }
354            _ => {}
355        }
356
357        // Various environment/ABIs are determined based on OS name.
358        match os {
359            "3ds" | "rtems" | "espidf" => env = "newlib",
360            "vxworks" => env = "gnu",
361            "redox" => env = "relibc",
362            "aix" => abi = "vec-extabi",
363            _ => {}
364        }
365
366        // Extra overrides for badly named targets.
367        match target {
368            // Actually simulator targets.
369            "i386-apple-ios" | "x86_64-apple-ios" | "x86_64-apple-tvos" => {
370                env = "sim";
371            }
372            // Name should've contained `muslabi64`.
373            "mips64-openwrt-linux-musl" => {
374                abi = "abi64";
375            }
376            // Specifies abi even though not in name.
377            "armv6-unknown-freebsd" | "armv6k-nintendo-3ds" | "armv7-unknown-freebsd" => {
378                abi = "eabihf";
379            }
380            // Specifies abi even though not in name.
381            "armv7-unknown-linux-ohos" | "armv7-unknown-trusty" => {
382                abi = "eabi";
383            }
384            _ => {}
385        }
386
387        let os = match os {
388            // Horizon is the common/internal OS name for 3DS and the Switch.
389            "3ds" | "switch" => "horizon",
390            // FIXME(madsmtm): macOS targets are badly named.
391            "darwin" => "macos",
392
393            // WASI targets contain the preview version in them too. Should've
394            // been `wasi-p1`/`wasi-p2`, but that's probably too late now.
395            os if os.starts_with("wasi") => {
396                env = os.strip_prefix("wasi").unwrap();
397                "wasi"
398            }
399            // FIXME(madsmtm): Badly named targets `*-linux-androideabi`,
400            // should be `*-android-eabi`.
401            "androideabi" => {
402                abi = "eabi";
403                "android"
404            }
405
406            os => os,
407        };
408
409        let vendor = match vendor {
410            // esp, esp32, esp32s2 etc.
411            vendor if vendor.starts_with("esp") => "espressif",
412            // FIXME(madsmtm): Badly named targets `*-linux-android*`,
413            // "linux" makes no sense as the vendor name.
414            "linux" if os == "android" || os == "androideabi" => "unknown",
415            // FIXME(madsmtm): Fix in `rustc` after
416            // https://github.com/rust-lang/compiler-team/issues/850.
417            "wali" => "unknown",
418            "lynx" => "unknown",
419            // Some Linux distributions set their name as the target vendor,
420            // so we have to assume that it can be an arbitary string.
421            vendor => vendor,
422        };
423
424        // Intentionally also marked as an ABI:
425        // https://github.com/rust-lang/rust/pull/86922
426        if vendor == "fortanix" {
427            abi = "fortanix";
428        }
429        if vendor == "uwp" {
430            abi = "uwp";
431        }
432        if ["powerpc64-unknown-linux-gnu", "powerpc64-wrs-vxworks"].contains(&target) {
433            abi = "elfv1";
434        }
435        if [
436            "powerpc64-unknown-freebsd",
437            "powerpc64-unknown-linux-musl",
438            "powerpc64-unknown-openbsd",
439            "powerpc64le-unknown-freebsd",
440            "powerpc64le-unknown-linux-gnu",
441            "powerpc64le-unknown-linux-musl",
442        ]
443        .contains(&target)
444        {
445            abi = "elfv2";
446        }
447
448        Ok(Self {
449            full_arch,
450            arch,
451            vendor,
452            os,
453            env,
454            abi,
455        })
456    }
457}

518    fn target_from_rustc_cfgs<'a>(target: &'a str, cfgs: &'a str) -> TargetInfo<'a> {
519        // Cannot determine full architecture from cfgs.
520        let (full_arch, _rest) = target.split_once('-').expect("target to have arch");
521
522        let mut target = TargetInfo {
523            full_arch,
524            arch: "invalid-none-set",
525            vendor: "invalid-none-set",
526            os: "invalid-none-set",
527            env: "invalid-none-set",
528            // Not set in older Rust versions
529            abi: "",
530        };
531
532        for cfg in cfgs.lines() {
533            if let Some((name, value)) = cfg.split_once('=') {
534                // Remove whitespace, if `rustc` decided to insert any.
535                let name = name.trim();
536                let value = value.trim();
537
538                // Remove quotes around value.
539                let value = value.strip_prefix('"').unwrap_or(value);
540                let value = value.strip_suffix('"').unwrap_or(value);
541
542                match name {
543                    "target_arch" => target.arch = value,
544                    "target_vendor" => target.vendor = value,
545                    "target_os" => target.os = value,
546                    "target_env" => target.env = value,
547                    "target_abi" => target.abi = value,
548                    _ => {}
549                }
550            } else {
551                // Skip cfgs like `debug_assertions` and `unix`.
552            }
553        }
554
555        if matches!(target.abi, "macabi" | "sim") {
556            assert_eq!(target.env, target.abi);
557            target.abi = "";
558        }
559
560        target
561    }

572        not(rustc_target_test),
573        ignore = "must enable explicitly with --cfg=rustc_target_test"

575    fn parse_rustc_targets() {
576        let rustc = std::env::var("RUSTC").unwrap_or_else(|_| "rustc".to_string());
577
578        let target_list = Command::new(&rustc)
579            .arg("--print=target-list")
580            .output()
581            .unwrap()
582            .stdout;
583        let target_list = String::from_utf8(target_list).unwrap();
584
585        let mut has_failure = false;
586        for target in target_list.lines() {
587            let cfgs = Command::new(&rustc)
588                .arg("--target")
589                .arg(target)
590                .arg("--print=cfg")
591                .output()
592                .unwrap()
593                .stdout;
594            let cfgs = String::from_utf8(cfgs).unwrap();
595
596            let expected = target_from_rustc_cfgs(target, &cfgs);
597            let actual = TargetInfo::from_rustc_target(target);
598
599            if Some(&expected) != actual.as_ref().ok() {
600                eprintln!("failed comparing {target}:");
601                eprintln!("  expected: Ok({expected:?})");
602                eprintln!("    actual: {actual:?}");
603                eprintln!();
604                has_failure = true;
605            }
606        }
607
608        if has_failure {
609            panic!("failed comparing targets");
610        }
611    }
