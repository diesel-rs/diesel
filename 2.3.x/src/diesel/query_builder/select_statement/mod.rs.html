95/// Semi-Private trait for containing get-functions for all `SelectStatement` fields
96//
97// This is used by `#[derive(MultiConnection)]`
98pub trait SelectStatementAccessor {
99    /// The type of the select clause
100    type Select;
101    /// The type of the from clause
102    type From;
103    /// The type of the distinct clause
104    type Distinct;
105    /// The type of the where clause
106    type Where;
107    /// The type of the order clause
108    type Order;
109    /// The type of the limit offset clause
110    type LimitOffset;
111    /// The type of the group by clause
112    type GroupBy;
113    /// The type of the having clause
114    type Having;
115    /// The type of the locking clause
116    type Locking;
117
118    /// Access the select clause
119    fn select_clause(&self) -> &Self::Select;
120    /// Access the from clause
121    #[allow(clippy::wrong_self_convention)] // obviously wrong, as `from` refers to the clause name
122    fn from_clause(&self) -> &Self::From;
123    /// Access the distinct clause
124    fn distinct_clause(&self) -> &Self::Distinct;
125    /// Access the where clause
126    fn where_clause(&self) -> &Self::Where;
127    /// Access the order clause
128    fn order_clause(&self) -> &Self::Order;
129    /// Access the limit_offset clause
130    fn limit_offset_clause(&self) -> &Self::LimitOffset;
131    /// Access the group by clause
132    fn group_by_clause(&self) -> &Self::GroupBy;
133    /// Access the having clause
134    fn having_clause(&self) -> &Self::Having;
135    /// Access the locking clause
136    fn locking_clause(&self) -> &Self::Locking;
137}

267impl<F, S, D, W, O, LOf, G, H, LC, DB>
268    QueryFragment<DB, sql_dialect::select_statement_syntax::AnsiSqlSelectStatement>
269    for SelectStatement<F, S, D, W, O, LOf, G, H, LC>
270where
271    DB: Backend<
272        SelectStatementSyntax = sql_dialect::select_statement_syntax::AnsiSqlSelectStatement,
273    >,
274    S: QueryFragment<DB>,
275    F: QueryFragment<DB>,
276    D: QueryFragment<DB>,
277    W: QueryFragment<DB>,
278    O: QueryFragment<DB>,
279    LOf: QueryFragment<DB>,
280    G: QueryFragment<DB>,
281    H: QueryFragment<DB>,
282    LC: QueryFragment<DB>,
283{
284    fn walk_ast<'b>(&'b self, mut out: AstPass<'_, 'b, DB>) -> QueryResult<()> {
285        out.push_sql("SELECT ");
286        self.distinct.walk_ast(out.reborrow())?;
287        self.select.walk_ast(out.reborrow())?;
288        self.from.walk_ast(out.reborrow())?;
289        self.where_clause.walk_ast(out.reborrow())?;
290        self.group_by.walk_ast(out.reborrow())?;
291        self.having.walk_ast(out.reborrow())?;
292        self.order.walk_ast(out.reborrow())?;
293        self.limit_offset.walk_ast(out.reborrow())?;
294        self.locking.walk_ast(out.reborrow())?;
295        Ok(())
296    }
297}
