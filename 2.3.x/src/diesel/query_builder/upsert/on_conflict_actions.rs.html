70{
71    fn walk_ast<'b>(&'b self, mut out: AstPass<'_, 'b, DB>) -> QueryResult<()> {
72        out.unsafe_to_cache_prepared();
73        if self.changeset.is_noop(out.backend())? {
74            out.push_sql(" DO NOTHING");
75        } else {
76            out.push_sql(" DO UPDATE SET ");

78        }
79        Ok(())
