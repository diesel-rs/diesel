34#[inline]
35pub(crate) unsafe fn read(fd: BorrowedFd<'_>, buf: (*mut u8, usize)) -> io::Result<usize> {
36    ret_usize(crate::backend::arch::choose::syscall3(crate::backend::reg::nr(linux_raw_sys::general::__NR_read),
    fd.into(), buf.0.into(), pass_usize(buf.1).into())syscall!(__NR_read, fd, buf.0, pass_usize(buf.1)))
37}

39#[inline]
40pub(crate) unsafe fn pread(
41    fd: BorrowedFd<'_>,
42    buf: (*mut u8, usize),
43    pos: u64,
44) -> io::Result<usize> {
45    // <https://git.kernel.org/pub/scm/linux/kernel/git/torvalds/linux.git/tree/arch/arm64/kernel/sys32.c?h=v6.13#n70>
46    #[cfg(all(
47        target_pointer_width = "32",
48        any(
49            target_arch = "arm",
50            target_arch = "mips",
51            target_arch = "mips32r6",
52            target_arch = "powerpc"
53        ),
54    ))]
55    {
56        ret_usize(syscall!(
57            __NR_pread64,
58            fd,
59            buf.0,
60            pass_usize(buf.1),
61            zero(),
62            hi(pos),
63            lo(pos)
64        ))
65    }
66    #[cfg(all(
67        target_pointer_width = "32",
68        not(any(
69            target_arch = "arm",
70            target_arch = "mips",
71            target_arch = "mips32r6",
72            target_arch = "powerpc"
73        )),
74    ))]
75    {
76        ret_usize(syscall!(
77            __NR_pread64,
78            fd,
79            buf.0,
80            pass_usize(buf.1),
81            hi(pos),
82            lo(pos)
83        ))
84    }
85    #[cfg(target_pointer_width = "64")]
86    ret_usize(crate::backend::arch::choose::syscall4(crate::backend::reg::nr(linux_raw_sys::general::__NR_pread64),
    fd.into(), buf.0.into(), pass_usize(buf.1).into(),
    loff_t_from_u64(pos).into())syscall!(
87        __NR_pread64,
88        fd,
89        buf.0,
90        pass_usize(buf.1),
91        loff_t_from_u64(pos)
92    ))
93}

102#[inline]
103pub(crate) fn preadv(
104    fd: BorrowedFd<'_>,
105    bufs: &mut [IoSliceMut<'_>],
106    pos: u64,
107) -> io::Result<usize> {
108    let (bufs_addr, bufs_len) = slice(&bufs[..cmp::min(bufs.len(), MAX_IOV)]);
109
110    // Unlike the plain "p" functions, the "pv" functions pass their offset in
111    // an endian-independent way, and always in two registers.
112    unsafe {
113        ret_usize(crate::backend::arch::choose::syscall5(crate::backend::reg::nr(linux_raw_sys::general::__NR_preadv),
    fd.into(), bufs_addr.into(), bufs_len.into(),
    pass_usize(pos as usize).into(), pass_usize((pos >> 32) as usize).into())syscall!(
114            __NR_preadv,
115            fd,
116            bufs_addr,
117            bufs_len,
118            pass_usize(pos as usize),
119            pass_usize((pos >> 32) as usize)
120        ))
121    }
122}

261pub(crate) unsafe fn close(fd: RawFd) {
262    // See the documentation for [`io::close`] for why errors are ignored.
263    crate::backend::arch::choose::syscall1_readonly(crate::backend::reg::nr(linux_raw_sys::general::__NR_close),
    raw_fd(fd).into())syscall_readonly!(__NR_close, raw_fd(fd)).decode_void();
264}

295#[allow(clippy::needless_pass_by_ref_mut)]
296#[inline]
297pub(crate) fn dup2(fd: BorrowedFd<'_>, new: &mut OwnedFd) -> io::Result<()> {
298    #[cfg(any(target_arch = "aarch64", target_arch = "riscv64"))]
299    {
300        // We don't need to worry about the difference between `dup2` and
301        // `dup3` when the file descriptors are equal because we have an
302        // `&mut OwnedFd` which means `fd` doesn't alias it.
303        dup3(fd, new, DupFlags::empty())
304    }
305
306    #[cfg(not(any(target_arch = "aarch64", target_arch = "riscv64")))]
307    unsafe {
308        ret_discarded_fd(crate::backend::arch::choose::syscall2_readonly(crate::backend::reg::nr(linux_raw_sys::general::__NR_dup2),
    fd.into(), new.as_fd().into())syscall_readonly!(__NR_dup2, fd, new.as_fd()))
309    }
310}

318#[inline]
319pub(crate) fn fcntl_getfd(fd: BorrowedFd<'_>) -> io::Result<FdFlags> {
320    #[cfg(target_pointer_width = "32")]
321    unsafe {
322        ret_c_uint(syscall_readonly!(__NR_fcntl64, fd, c_uint(F_GETFD)))
323            .map(FdFlags::from_bits_retain)
324    }
325    #[cfg(target_pointer_width = "64")]
326    unsafe {
327        ret_c_uint(crate::backend::arch::choose::syscall2_readonly(crate::backend::reg::nr(linux_raw_sys::general::__NR_fcntl),
    fd.into(), c_uint(F_GETFD).into())syscall_readonly!(__NR_fcntl, fd, c_uint(F_GETFD)))
328            .map(FdFlags::from_bits_retain)
329    }
330}
