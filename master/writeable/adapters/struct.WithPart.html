Struct WithPart

pub struct WithPart<T: ?Sized> {
    pub part: Part,
    pub writeable: T,
}
Expand description

A Writeable that writes out the given part.

§Examples

use writeable::adapters::WithPart;
use writeable::assert_writeable_parts_eq;
use writeable::Part;

// Simple usage:

const PART: Part = Part {
    category: "foo",
    value: "bar",
};

assert_writeable_parts_eq!(
    WithPart {
        writeable: "Hello World",
        part: PART
    },
    "Hello World",
    [(0, 11, PART)],
);

// Can be nested:

const PART2: Part = Part {
    category: "foo2",
    value: "bar2",
};

assert_writeable_parts_eq!(
    WithPart {
        writeable: WithPart {
            writeable: "Hello World",
            part: PART
        },
        part: PART2
    },
    "Hello World",
    [(0, 11, PART), (0, 11, PART2)],
);
