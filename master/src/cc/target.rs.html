12/// Information specific to a `rustc` target.
13///
14/// See <https://doc.rust-lang.org/cargo/appendix/glossary.html#target>.
15#[derive(Debug, PartialEq, Clone)]
16pub(crate) struct TargetInfo<'a> {
17    /// The full architecture, including the subarchitecture.
18    ///
19    /// This differs from `cfg!(target_arch)`, which only specifies the
20    /// overall architecture, which is too coarse for certain cases.
21    pub full_arch: &'a str,
22    /// The overall target architecture.
23    ///
24    /// This is the same as the value of `cfg!(target_arch)`.
25    pub arch: &'a str,
26    /// The target vendor.
27    ///
28    /// This is the same as the value of `cfg!(target_vendor)`.
29    pub vendor: &'a str,
30    /// The operating system, or `none` on bare-metal targets.
31    ///
32    /// This is the same as the value of `cfg!(target_os)`.
33    pub os: &'a str,
34    /// The environment on top of the operating system.
35    ///
36    /// This is the same as the value of `cfg!(target_env)`.
37    pub env: &'a str,
38    /// The ABI on top of the operating system.
39    ///
40    /// This is the same as the value of `cfg!(target_abi)`.
41    pub abi: &'a str,
42}
