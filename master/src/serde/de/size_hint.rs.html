11pub fn cautious<Element>(hint: Option<usize>) -> usize {
12    const MAX_PREALLOC_BYTES: usize = 1024 * 1024;
13
14    if mem::size_of::<Element>() == 0 {
15        0
16    } else {
17        cmp::min(
18            hint.unwrap_or(0),
19            MAX_PREALLOC_BYTES / mem::size_of::<Element>(),
20        )
21    }
22}
