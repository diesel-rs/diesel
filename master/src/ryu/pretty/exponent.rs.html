17        let d = DIGIT_TABLE.as_ptr().offset(k * 2);
18        ptr::copy_nonoverlapping(d, result.offset(1), 2);

20    } else if k >= 10 {
21        let d = DIGIT_TABLE.as_ptr().offset(k * 2);
22        ptr::copy_nonoverlapping(d, result, 2);
23        sign as usize + 2
24    } else {

27    }
