6#[cfg(all(feature = "mysql_backend", feature = "serde_json"))]
7impl FromSql<sql_types::Json, Mysql> for serde_json::Value {
8    fn from_sql(value: MysqlValue<'_>) -> deserialize::Result<Self> {
9        serde_json::from_slice(value.as_bytes()).map_err(|_| "Invalid Json".into())
10    }
11}
12
13#[cfg(all(feature = "mysql_backend", feature = "serde_json"))]
14impl ToSql<sql_types::Json, Mysql> for serde_json::Value {
15    fn to_sql<'b>(&'b self, out: &mut Output<'b, '_, Mysql>) -> serialize::Result {
16        serde_json::to_writer(out, self)
17            .map(|_| IsNull::No)
18            .map_err(Into::into)
19    }
20}

34    }

41            MysqlValue::new_internal(input_json, MysqlType::String),
42        )
