82/* sin(PI*x) assuming x > 2^-100, if sin(PI*x)==0 the sign is arbitrary */
83fn sin_pi(mut x: f32) -> f32 {
84    let mut y: f64;
85    let mut n: isize;
86
87    /* spurious inexact if odd int */
88    x = 2.0 * (x * 0.5 - floorf(x * 0.5)); /* x mod 2.0 */
89
90    n = (x * 4.0) as isize;
91    n = div!(n + 1, 2);
92    y = (x as f64) - (n as f64) * 0.5;
93    y *= 3.14159265358979323846;
94    match n {
95        1 => k_cosf(y),
96        2 => k_sinf(-y),
97        3 => -k_cosf(y),
98        // 0
99        _ => k_sinf(y),
100    }
101}

103#[cfg_attr(all(test, assert_no_panic), no_panic::no_panic)]
104pub fn lgammaf_r(mut x: f32) -> (f32, i32) {
105    let u = x.to_bits();
106    let mut t: f32;
107    let y: f32;
108    let mut z: f32;
109    let nadj: f32;
110    let p: f32;
111    let p1: f32;
112    let p2: f32;
113    let p3: f32;
114    let q: f32;
115    let mut r: f32;
116    let w: f32;
117    let ix: u32;
118    let i: i32;
119    let sign: bool;
120    let mut signgam: i32;
121
122    /* purge off +-inf, NaN, +-0, tiny and negative arguments */
123    signgam = 1;
124    sign = (u >> 31) != 0;
125    ix = u & 0x7fffffff;
126    if ix >= 0x7f800000 {
127        return (x * x, signgam);
128    }
129    if ix < 0x35000000 {
130        /* |x| < 2**-21, return -log(|x|) */
131        if sign {
132            signgam = -1;
133            x = -x;
134        }
135        return (-logf(x), signgam);
136    }
137    if sign {
138        x = -x;
139        t = sin_pi(x);
140        if t == 0.0 {
141            /* -integer */
142            return (1.0 / (x - x), signgam);
143        }
144        if t > 0.0 {
145            signgam = -1;
146        } else {
147            t = -t;
148        }
149        nadj = logf(PI / (t * x));
150    } else {
151        nadj = 0.0;
152    }
153
154    /* purge off 1 and 2 */
155    if ix == 0x3f800000 || ix == 0x40000000 {
156        r = 0.0;
157    }
158    /* for x < 2.0 */
159    else if ix < 0x40000000 {
160        if ix <= 0x3f666666 {
161            /* lgamma(x) = lgamma(x+1)-log(x) */
162            r = -logf(x);
163            if ix >= 0x3f3b4a20 {
164                y = 1.0 - x;
165                i = 0;
166            } else if ix >= 0x3e6d3308 {
167                y = x - (TC - 1.0);
168                i = 1;
169            } else {
170                y = x;
171                i = 2;
172            }
173        } else {
174            r = 0.0;
175            if ix >= 0x3fdda618 {
176                /* [1.7316,2] */
177                y = 2.0 - x;
178                i = 0;
179            } else if ix >= 0x3F9da620 {
180                /* [1.23,1.73] */
181                y = x - TC;
182                i = 1;
183            } else {
184                y = x - 1.0;
185                i = 2;
186            }
187        }
188        match i {
189            0 => {
190                z = y * y;
191                p1 = A0 + z * (A2 + z * (A4 + z * (A6 + z * (A8 + z * A10))));
192                p2 = z * (A1 + z * (A3 + z * (A5 + z * (A7 + z * (A9 + z * A11)))));
193                p = y * p1 + p2;
194                r += p - 0.5 * y;
195            }
196            1 => {
197                z = y * y;
198                w = z * y;
199                p1 = T0 + w * (T3 + w * (T6 + w * (T9 + w * T12))); /* parallel comp */
200                p2 = T1 + w * (T4 + w * (T7 + w * (T10 + w * T13)));
201                p3 = T2 + w * (T5 + w * (T8 + w * (T11 + w * T14)));
202                p = z * p1 - (TT - w * (p2 + y * p3));
203                r += TF + p;
204            }
205            2 => {
206                p1 = y * (U0 + y * (U1 + y * (U2 + y * (U3 + y * (U4 + y * U5)))));
207                p2 = 1.0 + y * (V1 + y * (V2 + y * (V3 + y * (V4 + y * V5))));
208                r += -0.5 * y + p1 / p2;
209            }
210            #[cfg(debug_assertions)]
211            _ => unreachable!(),
212            #[cfg(not(debug_assertions))]
213            _ => {}
214        }
215    } else if ix < 0x41000000 {
216        /* x < 8.0 */
217        i = x as i32;
218        y = x - (i as f32);
219        p = y * (S0 + y * (S1 + y * (S2 + y * (S3 + y * (S4 + y * (S5 + y * S6))))));
220        q = 1.0 + y * (R1 + y * (R2 + y * (R3 + y * (R4 + y * (R5 + y * R6)))));
221        r = 0.5 * y + p / q;
222        z = 1.0; /* lgamma(1+s) = log(s) + lgamma(s) */
223        // TODO: In C, this was implemented using switch jumps with fallthrough.
224        // Does this implementation have performance problems?
225        if i >= 7 {
226            z *= y + 6.0;
227        }
228        if i >= 6 {
229            z *= y + 5.0;
230        }
231        if i >= 5 {
232            z *= y + 4.0;
233        }
234        if i >= 4 {
235            z *= y + 3.0;
236        }
237        if i >= 3 {
238            z *= y + 2.0;
239            r += logf(z);
240        }
241    } else if ix < 0x5c800000 {
242        /* 8.0 <= x < 2**58 */
243        t = logf(x);
244        z = 1.0 / x;
245        y = z * z;
246        w = W0 + z * (W1 + y * (W2 + y * (W3 + y * (W4 + y * (W5 + y * W6)))));
247        r = (x - 0.5) * (t - 1.0) + w;
248    } else {
249        /* 2**58 <= x <= inf */
250        r = x * (logf(x) - 1.0);
251    }
252    if sign {
253        r = nadj - r;
254    }
255    return (r, signgam);
