15        pub const ERR_LIB_OFFSET: c_ulong = 23;

23        pub const fn ERR_SYSTEM_ERROR(errcode: c_ulong) -> bool {

25        }
26
27        pub const fn ERR_GET_LIB(errcode: c_ulong) -> c_int {
28            // hacks since `if` isn't yet stable in const functions :(
29            ((ERR_LIB_SYS as c_ulong * (ERR_SYSTEM_ERROR(errcode) as c_ulong)) |
30            (((errcode >> ERR_LIB_OFFSET) & ERR_LIB_MASK) * (!ERR_SYSTEM_ERROR(errcode) as c_ulong))) as c_int
31        }
32
33        pub const fn ERR_GET_FUNC(_errcode: c_ulong) -> c_int {
34            0
35        }

37        pub const fn ERR_GET_REASON(errcode: c_ulong) -> c_int {
38            // hacks since `if` isn't yet stable in const functions :(
39            ((ERR_LIB_SYS as c_ulong * (ERR_SYSTEM_ERROR(errcode) as c_ulong)) |
40            ((errcode & ERR_REASON_MASK) * (!ERR_SYSTEM_ERROR(errcode) as c_ulong))) as c_int
41        }
42
43        pub const fn ERR_PACK(lib: c_int, _func: c_int, reason: c_int) -> c_ulong {
44            ((lib as c_ulong & ERR_LIB_MASK) << ERR_LIB_OFFSET) |

46        }
47    } else {

49            ((l as c_ulong & 0x0FF) << 24) |
