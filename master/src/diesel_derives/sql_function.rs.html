19const VARIADIC_VARIANTS_DEFAULT: usize = 2;
20
21pub(crate) fn expand(
22    mut input: SqlFunctionDecl,
23    legacy_helper_type_and_module: bool,
24) -> TokenStream {
25    let attributes = &mut input.attributes;
26
27    let variadic_argument_count = attributes
28        .iter()
29        .find(|attr| attr.meta.path().is_ident("variadic"))
30        .map(|attr| {
31            let argument_count_literal = attr.parse_args::<VariadicAttributeArgs>()?.argument_count;
32            let argument_count = argument_count_literal.base10_parse::<usize>()?;
33
34            if argument_count > input.args.len() {
35                return Err(syn::Error::new(
36                    argument_count_literal.span(),
37                    "invalid variadic argument count: not enough function arguments",
38                ));
39            }
40
41            Ok(argument_count)
42        });
43
44    let Some(variadic_argument_count) = variadic_argument_count else {
45        let sql_name = parse_sql_name_attr(&mut input).unwrap_or_else(|| input.fn_name.to_string());
46
47        return expand_nonvariadic(input, sql_name, legacy_helper_type_and_module);
48    };
49
50    let variadic_argument_count = match variadic_argument_count {
51        Ok(arg_count) => arg_count,
52        Err(err) => return err.into_compile_error(),
53    };
54
55    attributes.retain(|attr| !attr.meta.path().is_ident("variadic"));
56
57    let variadic_variants = VARIADIC_VARIANTS_DEFAULT;
58
59    let mut result = TokenStream::new();
60    for variant_no in 0..=variadic_variants {
61        let expanded: TokenStream = expand_variadic(
62            input.clone(),
63            legacy_helper_type_and_module,
64            variadic_argument_count,
65            variant_no,
66        )
67        .unwrap_or_else(syn::Error::into_compile_error);
68
69        result.append_all(expanded.into_iter());
70    }
71
72    result
73}

75fn expand_variadic(
76    mut input: SqlFunctionDecl,
77    legacy_helper_type_and_module: bool,
78    variadic_argument_count: usize,
79    variant_no: usize,
80) -> syn::Result<TokenStream> {
81    add_variadic_doc_comments(&mut input.attributes, &input.fn_name.to_string());
82
83    let sql_name = parse_sql_name_attr(&mut input).unwrap_or_else(|| input.fn_name.to_string());
84
85    input.fn_name = format_ident!("{}_{}", input.fn_name, variant_no);
86
87    let nonvariadic_args_count = input.args.len() - variadic_argument_count;
88
89    let mut variadic_generic_indexes = vec![];
90    let mut arguments_with_generic_types = vec![];
91    for (arg_idx, arg) in input.args.iter().skip(nonvariadic_args_count).enumerate() {
92        // If argument is of type that definitely cannot be a generic then we skip it.
93        let Type::Path(ty_path) = arg.ty.clone() else {
94            continue;
95        };
96        let Ok(ty_ident) = ty_path.path.require_ident() else {
97            continue;
98        };
99
100        let idx = input.generics.params.iter().position(|param| match param {
101            GenericParam::Type(type_param) => type_param.ident == *ty_ident,
102            _ => false,
103        });
104
105        if let Some(idx) = idx {
106            variadic_generic_indexes.push(idx);
107            arguments_with_generic_types.push(arg_idx);
108        }
109    }
110
111    let mut args: Vec<_> = input.args.into_pairs().collect();
112    let variadic_args = args.split_off(nonvariadic_args_count);
113    let nonvariadic_args = args;
114
115    let variadic_args: Vec<_> = iter::repeat_n(variadic_args, variant_no)
116        .enumerate()
117        .flat_map(|(arg_group_idx, arg_group)| {
118            let mut resulting_args = vec![];
119
120            for (arg_idx, arg) in arg_group.into_iter().enumerate() {
121                let mut arg = arg.into_value();
122
123                arg.name = format_ident!("{}_{}", arg.name, arg_group_idx + 1);
124
125                if arguments_with_generic_types.contains(&arg_idx) {
126                    let Type::Path(mut ty_path) = arg.ty.clone() else {
127                        unreachable!("This argument should have path type as checked earlier")
128                    };
129                    let Ok(ident) = ty_path.path.require_ident() else {
130                        unreachable!("This argument should have ident type as checked earlier")
131                    };
132
133                    ty_path.path.segments[0].ident =
134                        format_ident!("{}{}", ident, arg_group_idx + 1);
135                    arg.ty = Type::Path(ty_path);
136                }
137
138                let pair = Pair::new(arg, Some(Token![,]([Span::call_site()])));
139                resulting_args.push(pair);
140            }
141
142            resulting_args
143        })
144        .collect();
145
146    input.args = nonvariadic_args.into_iter().chain(variadic_args).collect();
147
148    let generics: Vec<_> = input.generics.params.into_pairs().collect();
149    input.generics.params = if variant_no == 0 {
150        generics
151            .into_iter()
152            .enumerate()
153            .filter_map(|(generic_idx, generic)| {
154                (!variadic_generic_indexes.contains(&generic_idx)).then_some(generic)
155            })
156            .collect()
157    } else {
158        iter::repeat_n(generics, variant_no)
159            .enumerate()
160            .flat_map(|(generic_group_idx, generic_group)| {
161                let mut resulting_generics = vec![];
162
163                for (generic_idx, generic) in generic_group.into_iter().enumerate() {
164                    if !variadic_generic_indexes.contains(&generic_idx) {
165                        if generic_group_idx == 0 {
166                            resulting_generics.push(generic);
167                        }
168
169                        continue;
170                    }
171
172                    let mut generic = generic.into_value();
173
174                    if let GenericParam::Type(type_param) = &mut generic {
175                        type_param.ident =
176                            format_ident!("{}{}", type_param.ident, generic_group_idx + 1);
177                    } else {
178                        unreachable!("This generic should be a type param as checked earlier")
179                    }
180
181                    let pair = Pair::new(generic, Some(Token![,]([Span::call_site()])));
182                    resulting_generics.push(pair);
183                }
184
185                resulting_generics
186            })
187            .collect()
188    };
189
190    Ok(expand_nonvariadic(
191        input,
192        sql_name,
193        legacy_helper_type_and_module,
194    ))
195}

197fn add_variadic_doc_comments(attributes: &mut Vec<Attribute>, fn_name: &str) {
198    let mut doc_comments_end = attributes.len()
199        - attributes
200            .iter()
201            .rev()
202            .position(|attr| match &attr.meta {
203                Meta::NameValue(MetaNameValue { path, .. }) => path.is_ident("doc"),
204                _ => false,
205            })
206            .unwrap_or(attributes.len());
207
208    let fn_family = format!("`{0}_0`, `{0}_1`, ... `{0}_n`", fn_name);
209
210    let doc_comments: Vec<Attribute> = parse_quote! {
211        ///
212        /// # Variadic functions
213        ///
214        /// This function is variadic in SQL, so there's a family of functions
215        /// on a diesel side:
216        ///
217        #[doc = #fn_family]
218        ///
219        /// Here, the postfix number indicates repetitions of variadic arguments.
220        /// To use this function, the appropriate version with the correct
221        /// argument count must be selected.
222        #[doc(alias = #fn_name)]
223    };
224
225    for new_attribute in doc_comments {
226        attributes.insert(doc_comments_end, new_attribute);
227        doc_comments_end += 1;
228    }
229}

231fn parse_sql_name_attr(input: &mut SqlFunctionDecl) -> Option<String> {
232    let result = input
233        .attributes
234        .iter()
235        .find(|attr| attr.meta.path().is_ident("sql_name"))
236        .and_then(|attr| {
237            if let Meta::NameValue(MetaNameValue {
238                value:
239                    syn::Expr::Lit(syn::ExprLit {
240                        lit: syn::Lit::Str(ref lit),
241                        ..
242                    }),
243                ..
244            }) = attr.meta
245            {
246                Some(lit.value())
247            } else {
248                None
249            }
250        });
251
252    input
253        .attributes
254        .retain(|attr| !attr.meta.path().is_ident("sql_name"));
255
256    result
257}

259fn expand_nonvariadic(
260    input: SqlFunctionDecl,
261    sql_name: String,
262    legacy_helper_type_and_module: bool,
263) -> TokenStream {
264    let SqlFunctionDecl {
265        mut attributes,
266        fn_token,
267        fn_name,
268        mut generics,
269        args,
270        return_type,
271    } = input;
272
273    let is_aggregate = attributes
274        .iter()
275        .any(|attr| attr.meta.path().is_ident("aggregate"));
276
277    attributes.retain(|attr| !attr.meta.path().is_ident("aggregate"));
278
279    let args = &args;
280    let (ref arg_name, ref arg_type): (Vec<_>, Vec<_>) = args
281        .iter()
282        .map(|StrictFnArg { name, ty, .. }| (name, ty))
283        .unzip();
284    let arg_struct_assign = args.iter().map(
285        |StrictFnArg {
286             name, colon_token, ..
287         }| {
288            let name2 = name.clone();
289            quote!(#name #colon_token #name2.as_expression())
290        },
291    );
292
293    let type_args = &generics
294        .type_params()
295        .map(|type_param| type_param.ident.clone())
296        .collect::<Vec<_>>();
297
298    for StrictFnArg { name, .. } in args {
299        generics.params.push(parse_quote!(#name));
300    }
301
302    let (impl_generics, ty_generics, where_clause) = generics.split_for_impl();
303    // Even if we force an empty where clause, it still won't print the where
304    // token with no bounds.
305    let where_clause = where_clause
306        .map(|w| quote!(#w))
307        .unwrap_or_else(|| quote!(where));
308
309    let mut generics_with_internal = generics.clone();
310    generics_with_internal
311        .params
312        .push(parse_quote!(__DieselInternal));
313    let (impl_generics_internal, _, _) = generics_with_internal.split_for_impl();
314
315    let sql_type;
316    let numeric_derive;
317
318    if arg_name.is_empty() {
319        sql_type = None;
320        // FIXME: We can always derive once trivial bounds are stable
321        numeric_derive = None;
322    } else {
323        sql_type = Some(quote!((#(#arg_name),*): Expression,));
324        numeric_derive = Some(quote!(#[derive(diesel::sql_types::DieselNumericOps)]));
325    }
326
327    let helper_type_doc = format!("The return type of [`{fn_name}()`](super::fn_name)");
328
329    let args_iter = args.iter();
330    let mut tokens = quote! {
331        use diesel::{self, QueryResult};
332        use diesel::expression::{AsExpression, Expression, SelectableExpression, AppearsOnTable, ValidGrouping};
333        use diesel::query_builder::{QueryFragment, AstPass};
334        use diesel::sql_types::*;
335        use super::*;
336
337        #[derive(Debug, Clone, Copy, diesel::query_builder::QueryId)]
338        #numeric_derive
339        pub struct #fn_name #ty_generics {
340            #(pub(in super) #args_iter,)*
341            #(pub(in super) #type_args: ::std::marker::PhantomData<#type_args>,)*
342        }
343
344        #[doc = #helper_type_doc]
345        pub type HelperType #ty_generics = #fn_name <
346            #(#type_args,)*
347            #(<#arg_name as AsExpression<#arg_type>>::Expression,)*
348        >;
349
350        impl #impl_generics Expression for #fn_name #ty_generics
351        #where_clause
352            #sql_type
353        {
354            type SqlType = #return_type;
355        }
356
357        // __DieselInternal is what we call QS normally
358        impl #impl_generics_internal SelectableExpression<__DieselInternal>
359            for #fn_name #ty_generics
360        #where_clause
361            #(#arg_name: SelectableExpression<__DieselInternal>,)*
362            Self: AppearsOnTable<__DieselInternal>,
363        {
364        }
365
366        // __DieselInternal is what we call QS normally
367        impl #impl_generics_internal AppearsOnTable<__DieselInternal>
368            for #fn_name #ty_generics
369        #where_clause
370            #(#arg_name: AppearsOnTable<__DieselInternal>,)*
371            Self: Expression,
372        {
373        }
374
375        // __DieselInternal is what we call DB normally
376        impl #impl_generics_internal QueryFragment<__DieselInternal>
377            for #fn_name #ty_generics
378        where
379            __DieselInternal: diesel::backend::Backend,
380            #(#arg_name: QueryFragment<__DieselInternal>,)*
381        {
382            #[allow(unused_assignments)]
383            fn walk_ast<'__b>(&'__b self, mut out: AstPass<'_, '__b, __DieselInternal>) -> QueryResult<()>{
384                out.push_sql(concat!(#sql_name, "("));
385                // we unroll the arguments manually here, to prevent borrow check issues
386                let mut needs_comma = false;
387                #(
388                    if !self.#arg_name.is_noop(out.backend())? {
389                        if needs_comma {
390                            out.push_sql(", ");
391                        }
392                        self.#arg_name.walk_ast(out.reborrow())?;
393                        needs_comma = true;
394                    }
395                )*
396                out.push_sql(")");
397                Ok(())
398            }
399        }
400    };
401
402    let is_supported_on_sqlite = cfg!(feature = "sqlite")
403        && type_args.is_empty()
404        && is_sqlite_type(&return_type)
405        && arg_type.iter().all(|a| is_sqlite_type(a));
406
407    if is_aggregate {
408        tokens = quote! {
409            #tokens
410
411            impl #impl_generics_internal ValidGrouping<__DieselInternal>
412                for #fn_name #ty_generics
413            {
414                type IsAggregate = diesel::expression::is_aggregate::Yes;
415            }
416        };
417        if is_supported_on_sqlite {
418            tokens = quote! {
419                #tokens
420
421                use diesel::sqlite::{Sqlite, SqliteConnection};
422                use diesel::serialize::ToSql;
423                use diesel::deserialize::{FromSqlRow, StaticallySizedRow};
424                use diesel::sqlite::SqliteAggregateFunction;
425                use diesel::sql_types::IntoNullable;
426            };
427
428            match arg_name.len() {
429                x if x > 1 => {
430                    tokens = quote! {
431                        #tokens
432
433                        #[allow(dead_code)]
434                        /// Registers an implementation for this aggregate function on the given connection
435                        ///
436                        /// This function must be called for every `SqliteConnection` before
437                        /// this SQL function can be used on SQLite. The implementation must be
438                        /// deterministic (returns the same result given the same arguments).
439                        pub fn register_impl<A, #(#arg_name,)*>(
440                            conn: &mut SqliteConnection
441                        ) -> QueryResult<()>
442                            where
443                            A: SqliteAggregateFunction<(#(#arg_name,)*)>
444                                + Send
445                                + 'static
446                                + ::std::panic::UnwindSafe
447                                + ::std::panic::RefUnwindSafe,
448                            A::Output: ToSql<#return_type, Sqlite>,
449                            (#(#arg_name,)*): FromSqlRow<(#(#arg_type,)*), Sqlite> +
450                                StaticallySizedRow<(#(#arg_type,)*), Sqlite> +
451                                ::std::panic::UnwindSafe,
452                        {
453                            conn.register_aggregate_function::<(#(#arg_type,)*), #return_type, _, _, A>(#sql_name)
454                        }
455                    };
456                }
457                1 => {
458                    let arg_name = arg_name[0];
459                    let arg_type = arg_type[0];
460
461                    tokens = quote! {
462                        #tokens
463
464                        #[allow(dead_code)]
465                        /// Registers an implementation for this aggregate function on the given connection
466                        ///
467                        /// This function must be called for every `SqliteConnection` before
468                        /// this SQL function can be used on SQLite. The implementation must be
469                        /// deterministic (returns the same result given the same arguments).
470                        pub fn register_impl<A, #arg_name>(
471                            conn: &mut SqliteConnection
472                        ) -> QueryResult<()>
473                            where
474                            A: SqliteAggregateFunction<#arg_name>
475                                + Send
476                                + 'static
477                                + std::panic::UnwindSafe
478                                + std::panic::RefUnwindSafe,
479                            A::Output: ToSql<#return_type, Sqlite>,
480                            #arg_name: FromSqlRow<#arg_type, Sqlite> +
481                                StaticallySizedRow<#arg_type, Sqlite> +
482                                ::std::panic::UnwindSafe,
483                            {
484                                conn.register_aggregate_function::<#arg_type, #return_type, _, _, A>(#sql_name)
485                            }
486                    };
487                }
488                _ => (),
489            }
490        }
491    } else {
492        tokens = quote! {
493            #tokens
494
495            #[derive(ValidGrouping)]
496            pub struct __Derived<#(#arg_name,)*>(#(#arg_name,)*);
497
498            impl #impl_generics_internal ValidGrouping<__DieselInternal>
499                for #fn_name #ty_generics
500            where
501                __Derived<#(#arg_name,)*>: ValidGrouping<__DieselInternal>,
502            {
503                type IsAggregate = <__Derived<#(#arg_name,)*> as ValidGrouping<__DieselInternal>>::IsAggregate;
504            }
505        };
506
507        if is_supported_on_sqlite && !arg_name.is_empty() {
508            tokens = quote! {
509                #tokens
510
511                use diesel::sqlite::{Sqlite, SqliteConnection};
512                use diesel::serialize::ToSql;
513                use diesel::deserialize::{FromSqlRow, StaticallySizedRow};
514
515                #[allow(dead_code)]
516                /// Registers an implementation for this function on the given connection
517                ///
518                /// This function must be called for every `SqliteConnection` before
519                /// this SQL function can be used on SQLite. The implementation must be
520                /// deterministic (returns the same result given the same arguments). If
521                /// the function is nondeterministic, call
522                /// `register_nondeterministic_impl` instead.
523                pub fn register_impl<F, Ret, #(#arg_name,)*>(
524                    conn: &mut SqliteConnection,
525                    f: F,
526                ) -> QueryResult<()>
527                where
528                    F: Fn(#(#arg_name,)*) -> Ret + std::panic::UnwindSafe + Send + 'static,
529                    (#(#arg_name,)*): FromSqlRow<(#(#arg_type,)*), Sqlite> +
530                        StaticallySizedRow<(#(#arg_type,)*), Sqlite>,
531                    Ret: ToSql<#return_type, Sqlite>,
532                {
533                    conn.register_sql_function::<(#(#arg_type,)*), #return_type, _, _, _>(
534                        #sql_name,
535                        true,
536                        move |(#(#arg_name,)*)| f(#(#arg_name,)*),
537                    )
538                }
539
540                #[allow(dead_code)]
541                /// Registers an implementation for this function on the given connection
542                ///
543                /// This function must be called for every `SqliteConnection` before
544                /// this SQL function can be used on SQLite.
545                /// `register_nondeterministic_impl` should only be used if your
546                /// function can return different results with the same arguments (e.g.
547                /// `random`). If your function is deterministic, you should call
548                /// `register_impl` instead.
549                pub fn register_nondeterministic_impl<F, Ret, #(#arg_name,)*>(
550                    conn: &mut SqliteConnection,
551                    mut f: F,
552                ) -> QueryResult<()>
553                where
554                    F: FnMut(#(#arg_name,)*) -> Ret + std::panic::UnwindSafe + Send + 'static,
555                    (#(#arg_name,)*): FromSqlRow<(#(#arg_type,)*), Sqlite> +
556                        StaticallySizedRow<(#(#arg_type,)*), Sqlite>,
557                    Ret: ToSql<#return_type, Sqlite>,
558                {
559                    conn.register_sql_function::<(#(#arg_type,)*), #return_type, _, _, _>(
560                        #sql_name,
561                        false,
562                        move |(#(#arg_name,)*)| f(#(#arg_name,)*),
563                    )
564                }
565            };
566        }
567
568        if is_supported_on_sqlite && arg_name.is_empty() {
569            tokens = quote! {
570                #tokens
571
572                use diesel::sqlite::{Sqlite, SqliteConnection};
573                use diesel::serialize::ToSql;
574
575                #[allow(dead_code)]
576                /// Registers an implementation for this function on the given connection
577                ///
578                /// This function must be called for every `SqliteConnection` before
579                /// this SQL function can be used on SQLite. The implementation must be
580                /// deterministic (returns the same result given the same arguments). If
581                /// the function is nondeterministic, call
582                /// `register_nondeterministic_impl` instead.
583                pub fn register_impl<F, Ret>(
584                    conn: &SqliteConnection,
585                    f: F,
586                ) -> QueryResult<()>
587                where
588                    F: Fn() -> Ret + std::panic::UnwindSafe + Send + 'static,
589                    Ret: ToSql<#return_type, Sqlite>,
590                {
591                    conn.register_noarg_sql_function::<#return_type, _, _>(
592                        #sql_name,
593                        true,
594                        f,
595                    )
596                }
597
598                #[allow(dead_code)]
599                /// Registers an implementation for this function on the given connection
600                ///
601                /// This function must be called for every `SqliteConnection` before
602                /// this SQL function can be used on SQLite.
603                /// `register_nondeterministic_impl` should only be used if your
604                /// function can return different results with the same arguments (e.g.
605                /// `random`). If your function is deterministic, you should call
606                /// `register_impl` instead.
607                pub fn register_nondeterministic_impl<F, Ret>(
608                    conn: &SqliteConnection,
609                    mut f: F,
610                ) -> QueryResult<()>
611                where
612                    F: FnMut() -> Ret + std::panic::UnwindSafe + Send + 'static,
613                    Ret: ToSql<#return_type, Sqlite>,
614                {
615                    conn.register_noarg_sql_function::<#return_type, _, _>(
616                        #sql_name,
617                        false,
618                        f,
619                    )
620                }
621            };
622        }
623    }
624
625    let args_iter = args.iter();
626
627    let (outside_of_module_helper_type, return_type_path, internals_module_name) =
628        if legacy_helper_type_and_module {
629            (None, quote! { #fn_name::HelperType }, fn_name.clone())
630        } else {
631            let internals_module_name = Ident::new(&format!("{fn_name}_utils"), fn_name.span());
632            (
633                Some(quote! {
634                    #[allow(non_camel_case_types, non_snake_case)]
635                    #[doc = #helper_type_doc]
636                    pub type #fn_name #ty_generics = #internals_module_name::#fn_name <
637                        #(#type_args,)*
638                        #(<#arg_name as diesel::expression::AsExpression<#arg_type>>::Expression,)*
639                    >;
640                }),
641                quote! { #fn_name },
642                internals_module_name,
643            )
644        };
645
646    quote! {
647        #(#attributes)*
648        #[allow(non_camel_case_types)]
649        pub #fn_token #fn_name #impl_generics (#(#args_iter,)*)
650            -> #return_type_path #ty_generics
651        #where_clause
652            #(#arg_name: diesel::expression::AsExpression<#arg_type>,)*
653        {
654            #internals_module_name::#fn_name {
655                #(#arg_struct_assign,)*
656                #(#type_args: ::std::marker::PhantomData,)*
657            }
658        }
659
660        #outside_of_module_helper_type
661
662        #[doc(hidden)]
663        #[allow(non_camel_case_types, non_snake_case, unused_imports)]
664        pub(crate) mod #internals_module_name {
665            #tokens
666        }
667    }
668}

674impl Parse for ExternSqlBlock {
675    fn parse(input: ParseStream) -> Result<Self> {
676        let block = syn::ItemForeignMod::parse(input)?;
677        if block.abi.name.as_ref().map(|n| n.value()) != Some("SQL".into()) {
678            return Err(syn::Error::new(block.abi.span(), "expect `SQL` as ABI"));
679        }
680        if block.unsafety.is_some() {
681            return Err(syn::Error::new(
682                block.unsafety.unwrap().span(),
683                "expect `SQL` function blocks to be safe",
684            ));
685        }
686        let function_decls = block
687            .items
688            .into_iter()
689            .map(|i| syn::parse2(quote! { #i }))
690            .collect::<Result<Vec<_>>>()?;
691
692        Ok(ExternSqlBlock { function_decls })
693    }
694}

696#[derive(Clone)]
697pub(crate) struct SqlFunctionDecl {
698    attributes: Vec<Attribute>,
699    fn_token: Token![fn],
700    fn_name: Ident,
701    generics: Generics,
702    args: Punctuated<StrictFnArg, Token![,]>,
703    return_type: Type,
704}
705
706impl Parse for SqlFunctionDecl {
707    fn parse(input: ParseStream) -> Result<Self> {
708        let attributes = Attribute::parse_outer(input)?;
709        let fn_token: Token![fn] = input.parse()?;
710        let fn_name = Ident::parse(input)?;
711        let generics = Generics::parse(input)?;
712        let args;
713        let _paren = parenthesized!(args in input);
714        let args = args.parse_terminated(StrictFnArg::parse, Token![,])?;
715        let return_type = if Option::<Token![->]>::parse(input)?.is_some() {
716            Type::parse(input)?
717        } else {
718            parse_quote!(diesel::expression::expression_types::NotSelectable)
719        };
720        let _semi = Option::<Token![;]>::parse(input)?;
721
722        Ok(Self {
723            attributes,
724            fn_token,
725            fn_name,
726            generics,
727            args,
728            return_type,
729        })
730    }
731}

774fn is_sqlite_type(ty: &Type) -> bool {
775    let last_segment = if let Type::Path(tp) = ty {
776        if let Some(segment) = tp.path.segments.last() {
777            segment
778        } else {
779            return false;
780        }
781    } else {
782        return false;
783    };
784
785    let ident = last_segment.ident.to_string();
786    if ident == "Nullable" {
787        if let PathArguments::AngleBracketed(ref ab) = last_segment.arguments {
788            if let Some(GenericArgument::Type(ty)) = ab.args.first() {
789                return is_sqlite_type(ty);
790            }
791        }
792        return false;
793    }
794
795    [
796        "BigInt",
797        "Binary",
798        "Bool",
799        "Date",
800        "Double",
801        "Float",
802        "Integer",
803        "Numeric",
804        "SmallInt",
805        "Text",
806        "Time",
807        "Timestamp",
808    ]
809    .contains(&ident.as_str())
