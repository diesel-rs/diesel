1//! This crate provides Serde's two derive macros.
2//!
3//! ```edition2021
4//! # use serde_derive::{Deserialize, Serialize};
5//! #
6//! #[derive(Serialize, Deserialize)]
7//! # struct S;
8//! #
9//! # fn main() {}
10//! ```
11//!
12//! Please refer to [https://serde.rs/derive.html] for how to set this up.
13//!
14//! [https://serde.rs/derive.html]: https://serde.rs/derive.html

98impl private {
99    fn ident(&self) -> Ident {
100        Ident::new(
101            concat!("__private", env!("CARGO_PKG_VERSION_PATCH")),
102            Span::call_site(),
103        )
104    }
105}
106
107impl ToTokens for private {
108    fn to_tokens(&self, tokens: &mut proc_macro2::TokenStream) {
109        tokens.append(self.ident());
110    }
111}
112
113#[proc_macro_derive(Serialize, attributes(serde))]
114pub fn derive_serialize(input: TokenStream) -> TokenStream {
115    let mut input = parse_macro_input!(input as DeriveInput);
116    ser::expand_derive_serialize(&mut input)
117        .unwrap_or_else(syn::Error::into_compile_error)
118        .into()
119}
120
121#[proc_macro_derive(Deserialize, attributes(serde))]
122pub fn derive_deserialize(input: TokenStream) -> TokenStream {
123    let mut input = parse_macro_input!(input as DeriveInput);
124    de::expand_derive_deserialize(&mut input)
125        .unwrap_or_else(syn::Error::into_compile_error)
126        .into()
127}
